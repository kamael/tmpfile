121static void AppendUint32v(std::vector<byte>& buffer, uint32_t val) {
122 while (true) {
123 uint32_t next = val >> 7;
124 uint32_t out = val & 0x7f;
125 if (next) {
126 buffer.push_back(static_cast<byte>(0x80 | out));
127 val = next;
128 } else {
129 buffer.push_back(static_cast<byte>(out));
130 break;
131 }
132 }
