921// For DateTimePatternGenerator::addPatternWithSkeleton -
922// If skeletonToUse is specified, then an availableFormats entry is being added. In this case:
923// 1. We pass that skeleton to matcher.set instead of having it derive a skeleton from the pattern.
924// 2. If the new entry's skeleton or basePattern does match an existing entry but that entry also had a skeleton specified
925// (i.e. it was also from availableFormats), then the new entry does not override it regardless of the value of the override
926// parameter. This prevents later availableFormats entries from a parent locale overriding earlier ones from the actual
927// specified locale. However, availableFormats entries *should* override entries with matching skeleton whose skeleton was
928// derived (i.e. entries derived from the standard date/time patters for the specified locale).
929// 3. When adding the pattern (patternMap->add), we set a new boolean to indicate that the added entry had a
930// specified skeleton (which sets a new field in the PtnElem in the PatternMap).
931UDateTimePatternConflict
932DateTimePatternGenerator::addPatternWithSkeleton(
933 const UnicodeString& pattern,
934 const UnicodeString* skeletonToUse,
935 UBool override,
936 UnicodeString& conflictingPattern,
937 UErrorCode& status)
938{
939
940 UnicodeString basePattern;
941 PtnSkeleton skeleton;
942 UDateTimePatternConflict conflictingStatus = UDATPG_NO_CONFLICT;
943
944 DateTimeMatcher matcher;
945 if ( skeletonToUse == NULL__null ) {
946 matcher.set(pattern, fp, skeleton);
947 matcher.getBasePattern(basePattern);
948 } else {
949 matcher.set(*skeletonToUse, fp, skeleton); // no longer trims skeleton fields to max len 3, per #7930
950 matcher.getBasePattern(basePattern); // or perhaps instead: basePattern = *skeletonToUse;
951 }
952 // We only care about base conflicts - and replacing the pattern associated with a base - if:
953 // 1. the conflicting previous base pattern did *not* have an explicit skeleton; in that case the previous
954 // base + pattern combination was derived from either (a) a canonical item, (b) a standard format, or
955 // (c) a pattern specified programmatically with a previous call to addPattern (which would only happen
956 // if we are getting here from a subsequent call to addPattern).
957 // 2. a skeleton is specified for the current pattern, but override=false; in that case we are checking
958 // availableFormats items from root, which should not override any previous entry with the same base.
959 UBool entryHadSpecifiedSkeleton;
960 const UnicodeString *duplicatePattern = patternMap->getPatternFromBasePattern(basePattern, entryHadSpecifiedSkeleton);
961 if (duplicatePattern != NULL__null && (!entryHadSpecifiedSkeleton || (skeletonToUse != NULL__null && !override))) {
962 conflictingStatus = UDATPG_BASE_CONFLICT;
963 conflictingPattern = *duplicatePattern;
964 if (!override) {
965 return conflictingStatus;
966 }
967 }
968 // The only time we get here with override=true and skeletonToUse!=null is when adding availableFormats
969 // items from CLDR data. In that case, we don't want an item from a parent locale to replace an item with
970 // same skeleton from the specified locale, so skip the current item if skeletonWasSpecified is true for
971 // the previously-specified conflicting item.
972 const PtnSkeleton* entrySpecifiedSkeleton = NULL__null;
973 duplicatePattern = patternMap->getPatternFromSkeleton(skeleton, &entrySpecifiedSkeleton);
974 if (duplicatePattern != NULL__null ) {
975 conflictingStatus = UDATPG_CONFLICT;
976 conflictingPattern = *duplicatePattern;
977 if (!override || (skeletonToUse != NULL__null && entrySpecifiedSkeleton != NULL__null)) {
978 return conflictingStatus;
979 }
980 }
981 patternMap->add(basePattern, skeleton, pattern, skeletonToUse != NULL__null, status);
982 if(U_FAILURE(status)) {
983 return conflictingStatus;
984 }
985
986 return UDATPG_NO_CONFLICT;
987}

1070 }
1071 const dtTypeElem *row = &dtTypes[canonicalIndex];
1072 int32_t typeValue = row->field;
1073 if ((flags & kDTPGFixFractionalSeconds) != 0 && typeValue == UDATPG_SECOND_FIELD) {
1074 UnicodeString newField=dtMatcher->skeleton.original[UDATPG_FRACTIONAL_SECOND_FIELD];
1075 field = field + decimal + newField;
1076 } else if (dtMatcher->skeleton.type[typeValue]!=0) {
1077 // Here:
1078 // - "reqField" is the field from the originally requested skeleton, with length
1079 // "reqFieldLen".
1080 // - "field" is the field from the found pattern.
1081 //
1082 // The adjusted field should consist of characters from the originally requested
1083 // skeleton, except in the case of UDATPG_HOUR_FIELD or UDATPG_MONTH_FIELD or
1084 // UDATPG_WEEKDAY_FIELD or UDATPG_YEAR_FIELD, in which case it should consist
1085 // of characters from the found pattern.
1086 //
1087 // The length of the adjusted field (adjFieldLen) should match that in the originally
1088 // requested skeleton, except that in the following cases the length of the adjusted field
1089 // should match that in the found pattern (i.e. the length of this pattern field should
1090 // not be adjusted):
1091 // 1. typeValue is UDATPG_HOUR_FIELD/MINUTE/SECOND and the corresponding bit in options is
1092 // not set (ticket #7180). Note, we may want to implement a similar change for other
1093 // numeric fields (MM, dd, etc.) so the default behavior is to get locale preference for
1094 // field length, but options bits can be used to override this.
1095 // 2. There is a specified skeleton for the found pattern and one of the following is true:
1096 // a) The length of the field in the skeleton (skelFieldLen) is equal to reqFieldLen.
1097 // b) The pattern field is numeric and the skeleton field is not, or vice versa.
1098
1099 UnicodeString reqField = dtMatcher->skeleton.original[typeValue];
1100 int32_t reqFieldLen = reqField.length();
1101 if (reqField.charAt(0) == CAP_E((UChar)0x0045) && reqFieldLen < 3)
1102 reqFieldLen = 3; // 1-3 for E are equivalent to 3 for c,e
1103 int32_t adjFieldLen = reqFieldLen;
1104 if ( (typeValue==UDATPG_HOUR_FIELD && (options & UDATPG_MATCH_HOUR_FIELD_LENGTH)==0) ||
1105 (typeValue==UDATPG_MINUTE_FIELD && (options & UDATPG_MATCH_MINUTE_FIELD_LENGTH)==0) ||
1106 (typeValue==UDATPG_SECOND_FIELD && (options & UDATPG_MATCH_SECOND_FIELD_LENGTH)==0) ) {
1107 adjFieldLen = field.length();
1108 } else if (specifiedSkeleton) {
1109 UnicodeString skelField = specifiedSkeleton->original[typeValue];
1110 int32_t skelFieldLen = skelField.length();
1111 UBool patFieldIsNumeric = (row->type > 0);
1112 UBool skelFieldIsNumeric = (specifiedSkeleton->type[typeValue] > 0);
1113 if (skelFieldLen == reqFieldLen || (patFieldIsNumeric && !skelFieldIsNumeric) || (skelFieldIsNumeric && !patFieldIsNumeric)) {
1114 // don't adjust the field length in the found pattern
1115 adjFieldLen = field.length();
1116 }
1117 }
1118 UChar c = (typeValue!= UDATPG_HOUR_FIELD && typeValue!= UDATPG_MONTH_FIELD &&
1119 typeValue!= UDATPG_WEEKDAY_FIELD && (typeValue!= UDATPG_YEAR_FIELD || reqField.charAt(0)==CAP_Y((UChar)0x0059)))?
1120 reqField.charAt(0): field.charAt(0);
1121 if (typeValue == UDATPG_HOUR_FIELD && (flags & kDTPGSkeletonUsesCapJ) != 0) {
1122 c = fDefaultHourFormatChar;
1123 }
1124 field.remove();
1125 for (int32_t i=adjFieldLen; i>0; --i) {
1126 field+=c;
1127 }
1128 }
1129 newPattern+=field;
1130 }
1131 }
1132 return newPattern;
