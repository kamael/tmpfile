5233 U16_NEXT(patternString, patternStringIdx, patternStringLen, cPattern){ (cPattern)=(patternString)[(patternStringIdx)++]; if((((cPattern

)!=(patternStringLen) && (((__c2=(patternString)[(patternStringIdx
)])&0xfffffc00)==0xdc00)) { ++(patternStringIdx); (cPattern
)=(((UChar32)((cPattern))<<10UL)+(UChar32)(__c2)-((0xd800
<<10UL)+0xdc00-0x10000)); } } };

5235 if (cText != cPattern) {
5236 success = FALSE0;
5237 if (cText == U_SENTINEL(-1)) {
5238 fHitEnd = TRUE1;
5239 }
5240 break;
5241 }
5242 }

5244 success = FALSE0;
5245 }
5246
5247 if (success) {
5248 fp->fInputIdx = inputIterator.getIndex();
5249 } else {
5250 fp = (REStackFrame *)fStack->popFrame(fFrameSize);
5251 }
5252 }
5253 break;
5254
5255 case URX_LB_START:
5256 {
5257 // Entering a look-behind block.
5258 // Save Stack Ptr, Input Pos.
5259 // TODO: implement transparent bounds. Ticket #6067
5260 U_ASSERT(opValue>=0 && opValue+1<fPattern->fDataSize);
5261 fData[opValue] = fStack->size();

5263 // Init the variable containing the start index for attempted matches.
5264 fData[opValue+2] = -1;
5265 // Save input string length, then reset to pin any matches to end at
5266 // the current position.
5267 fData[opValue+3] = fActiveLimit;
5268 fActiveLimit = fp->fInputIdx;
5269 }
5270 break;
5271
5272
5273 case URX_LB_CONT:
5274 {
5275 // Positive Look-Behind, at top of loop checking for matches of LB expression
5276 // at all possible input starting positions.
5277
5278 // Fetch the min and max possible match lengths. They are the operands
5279 // of this op in the pattern.
5280 int32_t minML = (int32_t)pat[fp->fPatIdx++];
5281 int32_t maxML = (int32_t)pat[fp->fPatIdx++];
5282 U_ASSERT(minML <= maxML);
5283 U_ASSERT(minML >= 0);
5284
5285 // Fetch (from data) the last input index where a match was attempted.
5286 U_ASSERT(opValue>=0 && opValue+1<fPattern->fDataSize);
5287 int64_t *lbStartIdx = &fData[opValue+2];
5288 if (*lbStartIdx < 0) {
5289 // First time through loop.
5290 *lbStartIdx = fp->fInputIdx - minML;
5291 } else {
5292 // 2nd through nth time through the loop.
5293 // Back up start position for match by one.
5294 if (*lbStartIdx == 0) {
5295 (*lbStartIdx)--;
5296 } else {
5297 U16_BACK_1(inputBuf, 0, *lbStartIdx){ if(((((inputBuf)[--(*lbStartIdx)])&0xfffffc00)==0xdc00)
&& (*lbStartIdx)>(0) && ((((inputBuf)[(*lbStartIdx
)-1])&0xfffffc00)==0xd800)) { --(*lbStartIdx); } };
5298 }
5299 }
5300
5301 if (*lbStartIdx < 0 || *lbStartIdx < fp->fInputIdx - maxML) {
5302 // We have tried all potential match starting points without
5303 // getting a match. Backtrack out, and out of the
5304 // Look Behind altogether.
5305 fp = (REStackFrame *)fStack->popFrame(fFrameSize);
5306 int64_t restoreInputLen = fData[opValue+3];
5307 U_ASSERT(restoreInputLen >= fActiveLimit);
5308 U_ASSERT(restoreInputLen <= fInputLength);
5309 fActiveLimit = restoreInputLen;
5310 break;
5311 }
5312
5313 // Save state to this URX_LB_CONT op, so failure to match will repeat the loop.
5314 // (successful match will fall off the end of the loop.)
5315 fp = StateSave(fp, fp->fPatIdx-3, status);
5316 fp->fInputIdx = *lbStartIdx;
5317 }
5318 break;
5319
5320 case URX_LB_END:
5321 // End of a look-behind block, after a successful match.
5322 {
5323 U_ASSERT(opValue>=0 && opValue+1<fPattern->fDataSize);
5324 if (fp->fInputIdx != fActiveLimit) {
5325 // The look-behind expression matched, but the match did not
5326 // extend all the way to the point that we are looking behind from.
5327 // FAIL out of here, which will take us back to the LB_CONT, which
5328 // will retry the match starting at another position or fail
5329 // the look-behind altogether, whichever is appropriate.
5330 fp = (REStackFrame *)fStack->popFrame(fFrameSize);
5331 break;
5332 }
5333
5334 // Look-behind match is good. Restore the orignal input string length,
5335 // which had been truncated to pin the end of the lookbehind match to the
5336 // position being looked-behind.
5337 int64_t originalInputLen = fData[opValue+3];
5338 U_ASSERT(originalInputLen >= fActiveLimit);
5339 U_ASSERT(originalInputLen <= fInputLength);
5340 fActiveLimit = originalInputLen;
5341 }
5342 break;
5343
5344
5345 case URX_LBN_CONT:
5346 {
5347 // Negative Look-Behind, at top of loop checking for matches of LB expression
5348 // at all possible input starting positions.
5349
5350 // Fetch the extra parameters of this op.
5351 int32_t minML = (int32_t)pat[fp->fPatIdx++];
5352 int32_t maxML = (int32_t)pat[fp->fPatIdx++];
5353 int32_t continueLoc = (int32_t)pat[fp->fPatIdx++];
5354 continueLoc = URX_VAL(continueLoc)((continueLoc) & 0xffffff);
5355 U_ASSERT(minML <= maxML);
5356 U_ASSERT(minML >= 0);
5357 U_ASSERT(continueLoc > fp->fPatIdx);
5358
5359 // Fetch (from data) the last input index where a match was attempted.
5360 U_ASSERT(opValue>=0 && opValue+1<fPattern->fDataSize);
5361 int64_t *lbStartIdx = &fData[opValue+2];
5362 if (*lbStartIdx < 0) {
5363 // First time through loop.
5364 *lbStartIdx = fp->fInputIdx - minML;
5365 } else {
5366 // 2nd through nth time through the loop.
5367 // Back up start position for match by one.
5368 if (*lbStartIdx == 0) {
5369 (*lbStartIdx)--; // Because U16_BACK is unsafe starting at 0.
5370 } else {
5371 U16_BACK_1(inputBuf, 0, *lbStartIdx){ if(((((inputBuf)[--(*lbStartIdx)])&0xfffffc00)==0xdc00)
&& (*lbStartIdx)>(0) && ((((inputBuf)[(*lbStartIdx
)-1])&0xfffffc00)==0xd800)) { --(*lbStartIdx); } };
5372 }
5373 }
5374
5375 if (*lbStartIdx < 0 || *lbStartIdx < fp->fInputIdx - maxML) {
5376 // We have tried all potential match starting points without
5377 // getting a match, which means that the negative lookbehind as
5378 // a whole has succeeded. Jump forward to the continue location
5379 int64_t restoreInputLen = fData[opValue+3];
5380 U_ASSERT(restoreInputLen >= fActiveLimit);
5381 U_ASSERT(restoreInputLen <= fInputLength);
5382 fActiveLimit = restoreInputLen;
5383 fp->fPatIdx = continueLoc;
5384 break;
5385 }
5386
5387 // Save state to this URX_LB_CONT op, so failure to match will repeat the loop.
5388 // (successful match will cause a FAIL out of the loop altogether.)
5389 fp = StateSave(fp, fp->fPatIdx-4, status);
5390 fp->fInputIdx = *lbStartIdx;
5391 }
5392 break;
5393
5394 case URX_LBN_END:
5395 // End of a negative look-behind block, after a successful match.
5396 {
5397 U_ASSERT(opValue>=0 && opValue+1<fPattern->fDataSize);
5398 if (fp->fInputIdx != fActiveLimit) {
5399 // The look-behind expression matched, but the match did not
5400 // extend all the way to the point that we are looking behind from.
5401 // FAIL out of here, which will take us back to the LB_CONT, which
5402 // will retry the match starting at another position or succeed
5403 // the look-behind altogether, whichever is appropriate.
5404 fp = (REStackFrame *)fStack->popFrame(fFrameSize);
5405 break;
5406 }
5407
5408 // Look-behind expression matched, which means look-behind test as
5409 // a whole Fails
5410
5411 // Restore the orignal input string length, which had been truncated
5412 // inorder to pin the end of the lookbehind match
5413 // to the position being looked-behind.
5414 int64_t originalInputLen = fData[opValue+3];
5415 U_ASSERT(originalInputLen >= fActiveLimit);
5416 U_ASSERT(originalInputLen <= fInputLength);
5417 fActiveLimit = originalInputLen;
5418
5419 // Restore original stack position, discarding any state saved
5420 // by the successful pattern match.
5421 U_ASSERT(opValue>=0 && opValue+1<fPattern->fDataSize);
5422 int32_t newStackSize = (int32_t)fData[opValue];
5423 U_ASSERT(fStack->size() > newStackSize);
5424 fStack->setSize(newStackSize);
5425
5426 // FAIL, which will take control back to someplace
5427 // prior to entering the look-behind test.
5428 fp = (REStackFrame *)fStack->popFrame(fFrameSize);
5429 }
5430 break;
5431
5432
5433 case URX_LOOP_SR_I:
5434 // Loop Initialization for the optimized implementation of
5435 // [some character set]*
5436 // This op scans through all matching input.
5437 // The following LOOP_C op emulates stack unwinding if the following pattern fails.
5438 {
5439 U_ASSERT(opValue > 0 && opValue < sets->size());
5440 Regex8BitSet *s8 = &fPattern->fSets8[opValue];
5441 UnicodeSet *s = (UnicodeSet *)sets->elementAt(opValue);
5442
5443 // Loop through input, until either the input is exhausted or
5444 // we reach a character that is not a member of the set.
5445 int32_t ix = (int32_t)fp->fInputIdx;
5446 for (;;) {
5447 if (ix >= fActiveLimit) {
5448 fHitEnd = TRUE1;
5449 break;
5450 }
5451 UChar32 c;
5452 U16_NEXT(inputBuf, ix, fActiveLimit, c){ (c)=(inputBuf)[(ix)++]; if((((c)&0xfffffc00)==0xd800)) {

)[(ix)])&0xfffffc00)==0xdc00)) { ++(ix); (c)=(((UChar32)(
(c))<<10UL)+(UChar32)(__c2)-((0xd800<<10UL)+0xdc00
-0x10000)); } } };

5454 if (s8->contains(c) == FALSE0) {
5455 U16_BACK_1(inputBuf, 0, ix){ if(((((inputBuf)[--(ix)])&0xfffffc00)==0xdc00) &&
(ix)>(0) && ((((inputBuf)[(ix)-1])&0xfffffc00
)==0xd800)) { --(ix); } };
5456 break;
5457 }
5458 } else {
5459 if (s->contains(c) == FALSE0) {
5460 U16_BACK_1(inputBuf, 0, ix){ if(((((inputBuf)[--(ix)])&0xfffffc00)==0xdc00) &&
(ix)>(0) && ((((inputBuf)[(ix)-1])&0xfffffc00
)==0xd800)) { --(ix); } };
5461 break;
5462 }
5463 }
5464 }

5466 // If there were no matching characters, skip over the loop altogether.
5467 // The loop doesn't run at all, a * op always succeeds.
5468 if (ix == fp->fInputIdx) {
5469 fp->fPatIdx++; // skip the URX_LOOP_C op.
5470 break;
5471 }
5472
5473 // Peek ahead in the compiled pattern, to the URX_LOOP_C that
5474 // must follow. It's operand is the stack location
5475 // that holds the starting input index for the match of this [set]*
5476 int32_t loopcOp = (int32_t)pat[fp->fPatIdx];
5477 U_ASSERT(URX_TYPE(loopcOp) == URX_LOOP_C);

5479 U_ASSERT(stackLoc >= 0 && stackLoc < fFrameSize);

5483 // Save State to the URX_LOOP_C op that follows this one,

5487 fp->fPatIdx++;
5488 }

5498 // In DOTALL mode, we can just go straight to the end of the input.

5502 ix = (int32_t)fActiveLimit;

5536 // must follow. It's operand is the stack location
5537 // that holds the starting input index for the match of this .*

5554 case URX_LOOP_C:
5555 {
5556 U_ASSERT(opValue>=0 && opValue<fFrameSize);
5557 backSearchIndex = (int32_t)fp->fExtra[opValue];
5558 U_ASSERT(backSearchIndex <= fp->fInputIdx);

5561 // The loop is done. Leave here without saving state.
5562 // Subsequent failures won't come back here.
5563 break;

5566 // backed up by one from the last time through,

5568 // (We're going backwards because this loop emulates stack unwinding, not
5569 // the initial scan forward.)
5570 U_ASSERT(fp->fInputIdx > 0);
5571 UChar32 prevC;
5572 U16_PREV(inputBuf, 0, fp->fInputIdx, prevC){ (prevC)=(inputBuf)[--(fp->fInputIdx)]; if((((prevC)&

(0) && (((__c2=(inputBuf)[(fp->fInputIdx)-1])&

+0xdc00-0x10000)); } } }; // !!!: should this 0 be one of f*Limit?
5573

fInputIdx); } };
5581 }
5582 }
5583
5584
5585 fp = StateSave(fp, fp->fPatIdx-1, status);
5586 }
5587 break;

5593 // unrecognized type tag.
5594 U_ASSERT(FALSE);
5595 }
5596
5597 if (U_FAILURE(status)) {

5599 break;
5600 }
5601 }
