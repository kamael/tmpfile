1208bool BytecodeArrayBuilder::OperandIsValid(Bytecode bytecode, int operand_index,
1209 uint32_t operand_value) const {
1210 OperandType operand_type = Bytecodes::GetOperandType(bytecode, operand_index);
1211 switch (operand_type) {
1212 case OperandType::kNone:
1213 return false;
1214 case OperandType::kRegCount16: {
1215 // Expect kRegCount16 is part of a range previous operand is a
1216 // valid operand to start a range.
1217 if (operand_index > 0) {
1218 OperandType previous_operand_type =
1219 Bytecodes::GetOperandType(bytecode, operand_index - 1);
1220 return ((previous_operand_type == OperandType::kMaybeReg16 ||
1221 previous_operand_type == OperandType::kReg16) &&
1222 static_cast<uint16_t>(operand_value) == operand_value);
1223 } else {
1224 return false;
1225 }
1226 }
1227 case OperandType::kRegCount8: {
1228 // Expect kRegCount8 is part of a range previous operand is a
1229 // valid operand to start a range.
1230 if (operand_index > 0) {
1231 OperandType previous_operand_type =
1232 Bytecodes::GetOperandType(bytecode, operand_index - 1);
1233 return ((previous_operand_type == OperandType::kMaybeReg8 ||
1234 previous_operand_type == OperandType::kReg8 ||
1235 previous_operand_type == OperandType::kMaybeReg16) &&
1236 static_cast<uint8_t>(operand_value) == operand_value);
1237 } else {
1238 return false;
1239 }
1240 }
1241 case OperandType::kIdx16:
1242 return static_cast<uint16_t>(operand_value) == operand_value;
1243 case OperandType::kImm8:
1244 case OperandType::kIdx8:
1245 return static_cast<uint8_t>(operand_value) == operand_value;
1246 case OperandType::kMaybeReg8:
1247 if (operand_value == 0) {
1248 return true;
1249 }
1250 // Fall-through to kReg8 case.
1251 case OperandType::kReg8:
1252 case OperandType::kRegOut8:
1253 return RegisterIsValid(Register::FromRawOperand(operand_value),
1254 operand_type);
1255 case OperandType::kRegOutPair8:
1256 case OperandType::kRegOutPair16:
1257 case OperandType::kRegPair8:
1258 case OperandType::kRegPair16: {
1259 Register reg0 = Register::FromRawOperand(operand_value);
1260 Register reg1 = Register(reg0.index() + 1);
1261 return RegisterIsValid(reg0, operand_type) &&
1262 RegisterIsValid(reg1, operand_type);
1263 }
1264 case OperandType::kRegOutTriple8:
1265 case OperandType::kRegOutTriple16: {
1266 Register reg0 = Register::FromRawOperand(operand_value);
1267 Register reg1 = Register(reg0.index() + 1);
1268 Register reg2 = Register(reg0.index() + 2);
1269 return RegisterIsValid(reg0, operand_type) &&
1270 RegisterIsValid(reg1, operand_type) &&
1271 RegisterIsValid(reg2, operand_type);
1272 }
1273 case OperandType::kMaybeReg16:
1274 if (operand_value == 0) {
1275 return true;
1276 }
1277 // Fall-through to kReg16 case.
1278 case OperandType::kReg16:
1279 case OperandType::kRegOut16: {
1280 Register reg = Register::FromRawOperand(operand_value);
1281 return RegisterIsValid(reg, operand_type);
1282 }
1283 }
1284 UNREACHABLE()V8_Fatal("../src/interpreter/bytecode-array-builder.cc", 1284
, "unreachable code");
1285 return false;
1286}

1289bool BytecodeArrayBuilder::RegisterIsValid(Register reg,
1290 OperandType reg_type) const {
1291 if (!reg.is_valid()) {
1292 return false;
1293 }
1294
1295 switch (Bytecodes::SizeOfOperand(reg_type)) {
1296 case OperandSize::kByte:
1297 if (!FitsInReg8OperandUntranslated(reg)) {
1298 return false;
1299 }
1300 break;
1301 case OperandSize::kShort:
1302 if (!FitsInReg16OperandUntranslated(reg)) {
1303 return false;
1304 }
1305 break;
1306 case OperandSize::kNone:
1307 UNREACHABLE()V8_Fatal("../src/interpreter/bytecode-array-builder.cc", 1307
, "unreachable code");
1308 return false;
1309 }
1310
1311 if (reg.is_current_context() || reg.is_function_closure() ||
1312 reg.is_new_target()) {
1313 return true;
1314 } else if (reg.is_parameter()) {
1315 int parameter_index = reg.ToParameterIndex(parameter_count());
1316 return parameter_index >= 0 && parameter_index < parameter_count();
1317 } else if (RegisterTranslator::InTranslationWindow(reg)) {
1318 return translation_register_count() > 0;
1319 } else {
1320 reg = RegisterTranslator::UntranslateRegister(reg);
1321 if (reg.index() < fixed_register_count()) {
1322 return true;
1323 } else {
1324 return TemporaryRegisterIsLive(reg);
1325 }
1326 }
