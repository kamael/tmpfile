638UnicodeString&
639TimeZoneFormat::format(UTimeZoneFormatStyle style, const TimeZone& tz, UDate date,
640 UnicodeString& name, UTimeZoneFormatTimeType* timeType /* = NULL */) const {
641 if (timeType) {
642 *timeType = UTZFMT_TIME_TYPE_UNKNOWN;
643 }
644
645 UBool noOffsetFormatFallback = FALSE0;
646
647 switch (style) {
648 case UTZFMT_STYLE_GENERIC_LOCATION:
649 formatGeneric(tz, UTZGNM_LOCATION, date, name);
650 break;
651 case UTZFMT_STYLE_GENERIC_LONG:
652 formatGeneric(tz, UTZGNM_LONG, date, name);
653 break;
654 case UTZFMT_STYLE_GENERIC_SHORT:
655 formatGeneric(tz, UTZGNM_SHORT, date, name);
656 break;
657 case UTZFMT_STYLE_SPECIFIC_LONG:
658 formatSpecific(tz, UTZNM_LONG_STANDARD, UTZNM_LONG_DAYLIGHT, date, name, timeType);
659 break;
660 case UTZFMT_STYLE_SPECIFIC_SHORT:
661 formatSpecific(tz, UTZNM_SHORT_STANDARD, UTZNM_SHORT_DAYLIGHT, date, name, timeType);
662 break;
663
664 case UTZFMT_STYLE_ZONE_ID:
665 tz.getID(name);
666 noOffsetFormatFallback = TRUE1;
667 break;
668 case UTZFMT_STYLE_ZONE_ID_SHORT:
669 {
670 const UChar* shortID = ZoneMeta::getShortID(tz);
671 if (shortID == NULL__null) {
672 shortID = UNKNOWN_SHORT_ZONE_ID;
673 }
674 name.setTo(shortID, -1);
675 }
676 noOffsetFormatFallback = TRUE1;
677 break;
678
679 case UTZFMT_STYLE_EXEMPLAR_LOCATION:
680 formatExemplarLocation(tz, name);
681 noOffsetFormatFallback = TRUE1;
682 break;
683
684 default:
685 // will be handled below
686 break;
687 }
688
689 if (name.isEmpty() && !noOffsetFormatFallback) {
690 UErrorCode status = U_ZERO_ERROR;
691 int32_t rawOffset, dstOffset;
692 tz.getOffset(date, FALSE0, rawOffset, dstOffset, status);
693 int32_t offset = rawOffset + dstOffset;
694 if (U_SUCCESS(status)) {
695 switch (style) {
696 case UTZFMT_STYLE_GENERIC_LOCATION:
697 case UTZFMT_STYLE_GENERIC_LONG:
698 case UTZFMT_STYLE_SPECIFIC_LONG:
699 case UTZFMT_STYLE_LOCALIZED_GMT:
700 formatOffsetLocalizedGMT(offset, name, status);
701 break;
702
703 case UTZFMT_STYLE_GENERIC_SHORT:
704 case UTZFMT_STYLE_SPECIFIC_SHORT:
705 case UTZFMT_STYLE_LOCALIZED_GMT_SHORT:
706 formatOffsetShortLocalizedGMT(offset, name, status);
707 break;
708
709 case UTZFMT_STYLE_ISO_BASIC_SHORT:
710 formatOffsetISO8601Basic(offset, TRUE1, TRUE1, TRUE1, name, status);
711 break;
712
713 case UTZFMT_STYLE_ISO_BASIC_LOCAL_SHORT:
714 formatOffsetISO8601Basic(offset, FALSE0, TRUE1, TRUE1, name, status);
715 break;
716
717 case UTZFMT_STYLE_ISO_BASIC_FIXED:
718 formatOffsetISO8601Basic(offset, TRUE1, FALSE0, TRUE1, name, status);
719 break;
720
721 case UTZFMT_STYLE_ISO_BASIC_LOCAL_FIXED:
722 formatOffsetISO8601Basic(offset, FALSE0, FALSE0, TRUE1, name, status);
723 break;
724
725 case UTZFMT_STYLE_ISO_EXTENDED_FIXED:
726 formatOffsetISO8601Extended(offset, TRUE1, FALSE0, TRUE1, name, status);
727 break;
728
729 case UTZFMT_STYLE_ISO_EXTENDED_LOCAL_FIXED:
730 formatOffsetISO8601Extended(offset, FALSE0, FALSE0, TRUE1, name, status);
731 break;
732
733 case UTZFMT_STYLE_ISO_BASIC_FULL:
734 formatOffsetISO8601Basic(offset, TRUE1, FALSE0, FALSE0, name, status);
735 break;
736
737 case UTZFMT_STYLE_ISO_BASIC_LOCAL_FULL:
738 formatOffsetISO8601Basic(offset, FALSE0, FALSE0, FALSE0, name, status);
739 break;
740
741 case UTZFMT_STYLE_ISO_EXTENDED_FULL:
742 formatOffsetISO8601Extended(offset, TRUE1, FALSE0, FALSE0, name, status);
743 break;
744
745 case UTZFMT_STYLE_ISO_EXTENDED_LOCAL_FULL:
746 formatOffsetISO8601Extended(offset, FALSE0, FALSE0, FALSE0, name, status);
747 break;
748
749 default:
750 // UTZFMT_STYLE_ZONE_ID, UTZFMT_STYLE_ZONE_ID_SHORT, UTZFMT_STYLE_EXEMPLAR_LOCATION
751 break;
752 }
753
754 if (timeType) {
755 *timeType = (dstOffset != 0) ? UTZFMT_TIME_TYPE_DAYLIGHT : UTZFMT_TIME_TYPE_STANDARD;
756 }
757 }
758 }
759
760 return name;
761}

802
803TimeZone*
804TimeZoneFormat::parse(UTimeZoneFormatStyle style, const UnicodeString& text, ParsePosition& pos,
805 int32_t parseOptions, UTimeZoneFormatTimeType* timeType /* = NULL */) const {
806 if (timeType) {
807 *timeType = UTZFMT_TIME_TYPE_UNKNOWN;
808 }
809
810 int32_t startIdx = pos.getIndex();
811 int32_t maxPos = text.length();
812 int32_t offset;
813
814 // Styles using localized GMT format as fallback
815 UBool fallbackLocalizedGMT =
816 (style == UTZFMT_STYLE_SPECIFIC_LONG || style == UTZFMT_STYLE_GENERIC_LONG || style == UTZFMT_STYLE_GENERIC_LOCATION);
817 UBool fallbackShortLocalizedGMT =
818 (style == UTZFMT_STYLE_SPECIFIC_SHORT || style == UTZFMT_STYLE_GENERIC_SHORT);
819
820 int32_t evaluated = 0; // bit flags representing already evaluated styles
821 ParsePosition tmpPos(startIdx);
822
823 int32_t parsedOffset = UNKNOWN_OFFSET; // stores successfully parsed offset for later use
824 int32_t parsedPos = -1; // stores successfully parsed offset position for later use
825
826 // Try localized GMT format first if necessary
827 if (fallbackLocalizedGMT || fallbackShortLocalizedGMT) {
828 UBool hasDigitOffset = FALSE0;
829 offset = parseOffsetLocalizedGMT(text, tmpPos, fallbackShortLocalizedGMT, &hasDigitOffset);
830 if (tmpPos.getErrorIndex() == -1) {
831 // Even when the input text was successfully parsed as a localized GMT format text,
832 // we may still need to evaluate the specified style if -
833 // 1) GMT zero format was used, and
834 // 2) The input text was not completely processed
835 if (tmpPos.getIndex() == maxPos || hasDigitOffset) {
836 pos.setIndex(tmpPos.getIndex());
837 return createTimeZoneForOffset(offset);
838 }
839 parsedOffset = offset;
840 parsedPos = tmpPos.getIndex();
841 }
842 // Note: For now, no distinction between long/short localized GMT format in the parser.
843 // This might be changed in future.
844 // evaluated |= (fallbackLocalizedGMT ? STYLE_PARSE_FLAGS[UTZFMT_STYLE_LOCALIZED_GMT] : STYLE_PARSE_FLAGS[UTZFMT_STYLE_LOCALIZED_GMT_SHORT]);
845 evaluated |= STYLE_PARSE_FLAGS[UTZFMT_STYLE_LOCALIZED_GMT] | STYLE_PARSE_FLAGS[UTZFMT_STYLE_LOCALIZED_GMT_SHORT];
846 }
847
848 UErrorCode status = U_ZERO_ERROR;
849 UnicodeString tzID;
850
851 UBool parseTZDBAbbrev = ((parseOptions & UTZFMT_PARSE_OPTION_TZ_DATABASE_ABBREVIATIONS) != 0);
852
853 // Try the specified style
854 switch (style) {
855 case UTZFMT_STYLE_LOCALIZED_GMT:
856 {
857 tmpPos.setIndex(startIdx);
858 tmpPos.setErrorIndex(-1);
859
860 offset = parseOffsetLocalizedGMT(text, tmpPos);
861 if (tmpPos.getErrorIndex() == -1) {
862 pos.setIndex(tmpPos.getIndex());
863 return createTimeZoneForOffset(offset);
864 }
865
866 // Note: For now, no distinction between long/short localized GMT format in the parser.
867 // This might be changed in future.
868 evaluated |= STYLE_PARSE_FLAGS[UTZFMT_STYLE_LOCALIZED_GMT_SHORT];
869
870 break;
871 }
872 case UTZFMT_STYLE_LOCALIZED_GMT_SHORT:
873 {
874 tmpPos.setIndex(startIdx);
875 tmpPos.setErrorIndex(-1);
876
877 offset = parseOffsetShortLocalizedGMT(text, tmpPos);
878 if (tmpPos.getErrorIndex() == -1) {
879 pos.setIndex(tmpPos.getIndex());
880 return createTimeZoneForOffset(offset);
881 }
882
883 // Note: For now, no distinction between long/short localized GMT format in the parser.
884 // This might be changed in future.
885 evaluated |= STYLE_PARSE_FLAGS[UTZFMT_STYLE_LOCALIZED_GMT];
886
887 break;
888 }
889 case UTZFMT_STYLE_ISO_BASIC_SHORT:
890 case UTZFMT_STYLE_ISO_BASIC_FIXED:
891 case UTZFMT_STYLE_ISO_BASIC_FULL:
892 case UTZFMT_STYLE_ISO_EXTENDED_FIXED:
893 case UTZFMT_STYLE_ISO_EXTENDED_FULL:
894 {
895 tmpPos.setIndex(startIdx);
896 tmpPos.setErrorIndex(-1);
897
898 offset = parseOffsetISO8601(text, tmpPos);
899 if (tmpPos.getErrorIndex() == -1) {
900 pos.setIndex(tmpPos.getIndex());
901 return createTimeZoneForOffset(offset);
902 }
903
904 break;
905 }
906
907 case UTZFMT_STYLE_ISO_BASIC_LOCAL_SHORT:
908 case UTZFMT_STYLE_ISO_BASIC_LOCAL_FIXED:
909 case UTZFMT_STYLE_ISO_BASIC_LOCAL_FULL:
910 case UTZFMT_STYLE_ISO_EXTENDED_LOCAL_FIXED:
911 case UTZFMT_STYLE_ISO_EXTENDED_LOCAL_FULL:
912 {
913 tmpPos.setIndex(startIdx);
914 tmpPos.setErrorIndex(-1);
915
916 // Exclude the case of UTC Indicator "Z" here
917 UBool hasDigitOffset = FALSE0;
918 offset = parseOffsetISO8601(text, tmpPos, FALSE0, &hasDigitOffset);
919 if (tmpPos.getErrorIndex() == -1 && hasDigitOffset) {
920 pos.setIndex(tmpPos.getIndex());
921 return createTimeZoneForOffset(offset);
922 }
923
924 break;
925 }
926
927 case UTZFMT_STYLE_SPECIFIC_LONG:
928 case UTZFMT_STYLE_SPECIFIC_SHORT:
929 {
930 // Specific styles
931 int32_t nameTypes = 0;
932 if (style == UTZFMT_STYLE_SPECIFIC_LONG) {
933 nameTypes = (UTZNM_LONG_STANDARD | UTZNM_LONG_DAYLIGHT);
934 } else {
935 U_ASSERT(style == UTZFMT_STYLE_SPECIFIC_SHORT);
936 nameTypes = (UTZNM_SHORT_STANDARD | UTZNM_SHORT_DAYLIGHT);
937 }
938 LocalPointer<TimeZoneNames::MatchInfoCollection> specificMatches(fTimeZoneNames->find(text, startIdx, nameTypes, status));
939 if (U_FAILURE(status)) {
940 pos.setErrorIndex(startIdx);
941 return NULL__null;
942 }
943 if (!specificMatches.isNull()) {
944 int32_t matchIdx = -1;
945 int32_t matchPos = -1;
946 for (int32_t i = 0; i < specificMatches->size(); i++) {
947 matchPos = startIdx + specificMatches->getMatchLengthAt(i);
948 if (matchPos > parsedPos) {
949 matchIdx = i;
950 parsedPos = matchPos;
951 }
952 }
953 if (matchIdx >= 0) {
954 if (timeType) {
955 *timeType = getTimeType(specificMatches->getNameTypeAt(matchIdx));
956 }
957 pos.setIndex(matchPos);
958 getTimeZoneID(specificMatches.getAlias(), matchIdx, tzID);
959 U_ASSERT(!tzID.isEmpty());
960 return TimeZone::createTimeZone(tzID);
961 }
962 }
963
964 if (parseTZDBAbbrev && style == UTZFMT_STYLE_SPECIFIC_SHORT) {
965 U_ASSERT((nameTypes & UTZNM_SHORT_STANDARD) != 0);
966 U_ASSERT((nameTypes & UTZNM_SHORT_DAYLIGHT) != 0);
967
968 const TZDBTimeZoneNames *tzdbTimeZoneNames = getTZDBTimeZoneNames(status);
969 if (U_SUCCESS(status)) {
970 LocalPointer<TimeZoneNames::MatchInfoCollection> tzdbNameMatches(
971 tzdbTimeZoneNames->find(text, startIdx, nameTypes, status));
972 if (U_FAILURE(status)) {
973 pos.setErrorIndex(startIdx);
974 return NULL__null;
975 }
976 if (!tzdbNameMatches.isNull()) {
977 int32_t matchIdx = -1;
978 int32_t matchPos = -1;
979 for (int32_t i = 0; i < tzdbNameMatches->size(); i++) {
980 matchPos = startIdx + tzdbNameMatches->getMatchLengthAt(i);
981 if (matchPos > parsedPos) {
982 matchIdx = i;
983 parsedPos = matchPos;
984 }
985 }
986 if (matchIdx >= 0) {
987 if (timeType) {
988 *timeType = getTimeType(tzdbNameMatches->getNameTypeAt(matchIdx));
989 }
990 pos.setIndex(matchPos);
991 getTimeZoneID(tzdbNameMatches.getAlias(), matchIdx, tzID);
992 U_ASSERT(!tzID.isEmpty());
993 return TimeZone::createTimeZone(tzID);
994 }
995 }
996 }
997 }
998 break;
999 }
1000 case UTZFMT_STYLE_GENERIC_LONG:
1001 case UTZFMT_STYLE_GENERIC_SHORT:
1002 case UTZFMT_STYLE_GENERIC_LOCATION:
1003 {
1004 int32_t genericNameTypes = 0;
1005 switch (style) {
1006 case UTZFMT_STYLE_GENERIC_LOCATION:
1007 genericNameTypes = UTZGNM_LOCATION;
1008 break;
1009
1010 case UTZFMT_STYLE_GENERIC_LONG:
1011 genericNameTypes = UTZGNM_LONG | UTZGNM_LOCATION;
1012 break;
1013
1014 case UTZFMT_STYLE_GENERIC_SHORT:
1015 genericNameTypes = UTZGNM_SHORT | UTZGNM_LOCATION;
1016 break;
1017
1018 default:
1019 U_ASSERT(FALSE);
1020 }
1021
1022 int32_t len = 0;
1023 UTimeZoneFormatTimeType tt = UTZFMT_TIME_TYPE_UNKNOWN;
1024 const TimeZoneGenericNames *gnames = getTimeZoneGenericNames(status);
1025 if (U_SUCCESS(status)) {
1026 len = gnames->findBestMatch(text, startIdx, genericNameTypes, tzID, tt, status);
1027 }
1028 if (U_FAILURE(status)) {
1029 pos.setErrorIndex(startIdx);
1030 return NULL__null;
1031 }
1032 if (len > 0) {
1033 // Found a match
1034 if (timeType) {
1035 *timeType = tt;
1036 }
1037 pos.setIndex(startIdx + len);
1038 U_ASSERT(!tzID.isEmpty());
1039 return TimeZone::createTimeZone(tzID);
1040 }
1041
1042 break;
1043 }
1044 case UTZFMT_STYLE_ZONE_ID:
1045 {
1046 tmpPos.setIndex(startIdx);
1047 tmpPos.setErrorIndex(-1);
1048
1049 parseZoneID(text, tmpPos, tzID);
1050 if (tmpPos.getErrorIndex() == -1) {
1051 pos.setIndex(tmpPos.getIndex());
1052 return TimeZone::createTimeZone(tzID);
1053 }
1054 break;
1055 }
1056 case UTZFMT_STYLE_ZONE_ID_SHORT:
1057 {
1058 tmpPos.setIndex(startIdx);
1059 tmpPos.setErrorIndex(-1);
1060
1061 parseShortZoneID(text, tmpPos, tzID);
1062 if (tmpPos.getErrorIndex() == -1) {
1063 pos.setIndex(tmpPos.getIndex());
1064 return TimeZone::createTimeZone(tzID);
1065 }
1066 break;
1067 }
1068 case UTZFMT_STYLE_EXEMPLAR_LOCATION:
1069 {
1070 tmpPos.setIndex(startIdx);
1071 tmpPos.setErrorIndex(-1);
1072
1073 parseExemplarLocation(text, tmpPos, tzID);
1074 if (tmpPos.getErrorIndex() == -1) {
1075 pos.setIndex(tmpPos.getIndex());
1076 return TimeZone::createTimeZone(tzID);
1077 }
1078 break;
1079 }
1080 }
1081 evaluated |= STYLE_PARSE_FLAGS[style];
1082
1083
1084 if (parsedPos > startIdx) {
1085 // When the specified style is one of SPECIFIC_XXX or GENERIC_XXX, we tried to parse the input
1086 // as localized GMT format earlier. If parsedOffset is positive, it means it was successfully
1087 // parsed as localized GMT format, but offset digits were not detected (more specifically, GMT
1088 // zero format). Then, it tried to find a match within the set of display names, but could not
1089 // find a match. At this point, we can safely assume the input text contains the localized
1090 // GMT format.
1091 U_ASSERT(parsedOffset != UNKNOWN_OFFSET);
1092 pos.setIndex(parsedPos);
1093 return createTimeZoneForOffset(parsedOffset);
1094 }
1095
1096 // Failed to parse the input text as the time zone format in the specified style.
1097 // Check the longest match among other styles below.
1098 UnicodeString parsedID;
1099 UTimeZoneFormatTimeType parsedTimeType = UTZFMT_TIME_TYPE_UNKNOWN;
1100
1101 U_ASSERT(parsedPos < 0);
1102 U_ASSERT(parsedOffset == UNKNOWN_OFFSET);
1103
1104 // ISO 8601
1105 if (parsedPos < maxPos &&
1106 ((evaluated & ISO_Z_STYLE_FLAG0x0080) == 0 || (evaluated & ISO_LOCAL_STYLE_FLAG0x0100) == 0)) {
1107 tmpPos.setIndex(startIdx);
1108 tmpPos.setErrorIndex(-1);
1109
1110 UBool hasDigitOffset = FALSE0;
1111 offset = parseOffsetISO8601(text, tmpPos, FALSE0, &hasDigitOffset);
1112 if (tmpPos.getErrorIndex() == -1) {
1113 if (tmpPos.getIndex() == maxPos || hasDigitOffset) {
1114 pos.setIndex(tmpPos.getIndex());
1115 return createTimeZoneForOffset(offset);
1116 }
1117 // Note: When ISO 8601 format contains offset digits, it should not
1118 // collide with other formats. However, ISO 8601 UTC format "Z" (single letter)
1119 // may collide with other names. In this case, we need to evaluate other names.
1120 if (parsedPos < tmpPos.getIndex()) {
1121 parsedOffset = offset;
1122 parsedID.setToBogus();
1123 parsedTimeType = UTZFMT_TIME_TYPE_UNKNOWN;
1124 parsedPos = tmpPos.getIndex();
1125 U_ASSERT(parsedPos == startIdx + 1); // only when "Z" is used
1126 }
1127 }
1128 }
1129
1130 // Localized GMT format
1131 if (parsedPos < maxPos &&
1132 (evaluated & STYLE_PARSE_FLAGS[UTZFMT_STYLE_LOCALIZED_GMT]) == 0) {
1133 tmpPos.setIndex(startIdx);
1134 tmpPos.setErrorIndex(-1);
1135
1136 UBool hasDigitOffset = FALSE0;
1137 offset = parseOffsetLocalizedGMT(text, tmpPos, FALSE0, &hasDigitOffset);
1138 if (tmpPos.getErrorIndex() == -1) {
1139 if (tmpPos.getIndex() == maxPos || hasDigitOffset) {
1140 pos.setIndex(tmpPos.getIndex());
1141 return createTimeZoneForOffset(offset);
1142 }
1143 // Evaluate other names - see the comment earlier in this method.
1144 if (parsedPos < tmpPos.getIndex()) {
1145 parsedOffset = offset;
1146 parsedID.setToBogus();
1147 parsedTimeType = UTZFMT_TIME_TYPE_UNKNOWN;
1148 parsedPos = tmpPos.getIndex();
1149 }
1150 }
1151 }
1152
1153 if (parsedPos < maxPos &&
1154 (evaluated & STYLE_PARSE_FLAGS[UTZFMT_STYLE_LOCALIZED_GMT_SHORT]) == 0) {
1155 tmpPos.setIndex(startIdx);
1156 tmpPos.setErrorIndex(-1);
1157
1158 UBool hasDigitOffset = FALSE0;
1159 offset = parseOffsetLocalizedGMT(text, tmpPos, TRUE1, &hasDigitOffset);
1160 if (tmpPos.getErrorIndex() == -1) {
1161 if (tmpPos.getIndex() == maxPos || hasDigitOffset) {
1162 pos.setIndex(tmpPos.getIndex());
1163 return createTimeZoneForOffset(offset);
1164 }
1165 // Evaluate other names - see the comment earlier in this method.
1166 if (parsedPos < tmpPos.getIndex()) {
1167 parsedOffset = offset;
1168 parsedID.setToBogus();
1169 parsedTimeType = UTZFMT_TIME_TYPE_UNKNOWN;
1170 parsedPos = tmpPos.getIndex();
1171 }
1172 }
1173 }
1174
1175 // When ParseOption.ALL_STYLES is available, we also try to look all possible display names and IDs.
1176 // For example, when style is GENERIC_LONG, "EST" (SPECIFIC_SHORT) is never
1177 // used for America/New_York. With parseAllStyles true, this code parses "EST"
1178 // as America/New_York.
1179
1180 // Note: Adding all possible names into the trie used by the implementation is quite heavy operation,
1181 // which we want to avoid normally (note that we cache the trie, so this is applicable to the
1182 // first time only as long as the cache does not expire).
1183
1184 if (parseOptions & UTZFMT_PARSE_OPTION_ALL_STYLES) {
1185 // Try all specific names and exemplar location names
1186 if (parsedPos < maxPos) {
1187 LocalPointer<TimeZoneNames::MatchInfoCollection> specificMatches(fTimeZoneNames->find(text, startIdx, ALL_SIMPLE_NAME_TYPES, status));
1188 if (U_FAILURE(status)) {
1189 pos.setErrorIndex(startIdx);
1190 return NULL__null;
1191 }
1192 int32_t specificMatchIdx = -1;
1193 int32_t matchPos = -1;
1194 if (!specificMatches.isNull()) {
1195 for (int32_t i = 0; i < specificMatches->size(); i++) {
1196 if (startIdx + specificMatches->getMatchLengthAt(i) > matchPos) {
1197 specificMatchIdx = i;
1198 matchPos = startIdx + specificMatches->getMatchLengthAt(i);
1199 }
1200 }
1201 }
1202 if (parsedPos < matchPos) {
1203 U_ASSERT(specificMatchIdx >= 0);
1204 parsedPos = matchPos;
1205 getTimeZoneID(specificMatches.getAlias(), specificMatchIdx, parsedID);
1206 parsedTimeType = getTimeType(specificMatches->getNameTypeAt(specificMatchIdx));
1207 parsedOffset = UNKNOWN_OFFSET;
1208 }
1209 }
1210 if (parseTZDBAbbrev && parsedPos < maxPos && (evaluated & STYLE_PARSE_FLAGS[UTZFMT_STYLE_SPECIFIC_SHORT]) == 0) {
1211 const TZDBTimeZoneNames *tzdbTimeZoneNames = getTZDBTimeZoneNames(status);
1212 if (U_SUCCESS(status)) {
1213 LocalPointer<TimeZoneNames::MatchInfoCollection> tzdbNameMatches(
1214 tzdbTimeZoneNames->find(text, startIdx, ALL_SIMPLE_NAME_TYPES, status));
1215 if (U_FAILURE(status)) {
1216 pos.setErrorIndex(startIdx);
1217 return NULL__null;
1218 }
1219 int32_t tzdbNameMatchIdx = -1;
1220 int32_t matchPos = -1;
1221 if (!tzdbNameMatches.isNull()) {
1222 for (int32_t i = 0; i < tzdbNameMatches->size(); i++) {
1223 if (startIdx + tzdbNameMatches->getMatchLengthAt(i) > matchPos) {
1224 tzdbNameMatchIdx = i;
1225 matchPos = startIdx + tzdbNameMatches->getMatchLengthAt(i);
1226 }
1227 }
1228 }
1229 if (parsedPos < matchPos) {
1230 U_ASSERT(tzdbNameMatchIdx >= 0);
1231 parsedPos = matchPos;
1232 getTimeZoneID(tzdbNameMatches.getAlias(), tzdbNameMatchIdx, parsedID);
1233 parsedTimeType = getTimeType(tzdbNameMatches->getNameTypeAt(tzdbNameMatchIdx));
1234 parsedOffset = UNKNOWN_OFFSET;
1235 }
1236 }
1237 }
1238 // Try generic names
1239 if (parsedPos < maxPos) {
1240 int32_t genMatchLen = -1;
1241 UTimeZoneFormatTimeType tt = UTZFMT_TIME_TYPE_UNKNOWN;
1242
1243 const TimeZoneGenericNames *gnames = getTimeZoneGenericNames(status);
1244 if (U_SUCCESS(status)) {
1245 genMatchLen = gnames->findBestMatch(text, startIdx, ALL_GENERIC_NAME_TYPES, tzID, tt, status);
1246 }
1247 if (U_FAILURE(status)) {
1248 pos.setErrorIndex(startIdx);
1249 return NULL__null;
1250 }
1251
1252 if (genMatchLen > 0 && parsedPos < startIdx + genMatchLen) {
1253 parsedPos = startIdx + genMatchLen;
1254 parsedID.setTo(tzID);
1255 parsedTimeType = tt;
1256 parsedOffset = UNKNOWN_OFFSET;
1257 }
1258 }
1259
1260 // Try time zone ID
1261 if (parsedPos < maxPos && (evaluated & STYLE_PARSE_FLAGS[UTZFMT_STYLE_ZONE_ID]) == 0) {
1262 tmpPos.setIndex(startIdx);
1263 tmpPos.setErrorIndex(-1);
1264
1265 parseZoneID(text, tmpPos, tzID);
1266 if (tmpPos.getErrorIndex() == -1 && parsedPos < tmpPos.getIndex()) {
1267 parsedPos = tmpPos.getIndex();
1268 parsedID.setTo(tzID);
1269 parsedTimeType = UTZFMT_TIME_TYPE_UNKNOWN;
1270 parsedOffset = UNKNOWN_OFFSET;
1271 }
1272 }
1273 // Try short time zone ID
1274 if (parsedPos < maxPos && (evaluated & STYLE_PARSE_FLAGS[UTZFMT_STYLE_ZONE_ID]) == 0) {
1275 tmpPos.setIndex(startIdx);
1276 tmpPos.setErrorIndex(-1);
1277
1278 parseShortZoneID(text, tmpPos, tzID);
1279 if (tmpPos.getErrorIndex() == -1 && parsedPos < tmpPos.getIndex()) {
1280 parsedPos = tmpPos.getIndex();
1281 parsedID.setTo(tzID);
1282 parsedTimeType = UTZFMT_TIME_TYPE_UNKNOWN;
1283 parsedOffset = UNKNOWN_OFFSET;
1284 }
1285 }
1286 }
1287
1288 if (parsedPos > startIdx) {
1289 // Parsed successfully
1290 TimeZone* parsedTZ;
1291 if (parsedID.length() > 0) {
1292 parsedTZ = TimeZone::createTimeZone(parsedID);
1293 } else {
1294 U_ASSERT(parsedOffset != UNKNOWN_OFFSET);
1295 parsedTZ = createTimeZoneForOffset(parsedOffset);
1296 }
1297 if (timeType) {
1298 *timeType = parsedTimeType;
1299 }
1300 pos.setIndex(parsedPos);
1301 return parsedTZ;
1302 }
1303
1304 pos.setErrorIndex(startIdx);
1305 return NULL__null;
1306}

1469// Private zone offset format/parse implementation
1470
1471UnicodeString&
1472TimeZoneFormat::formatOffsetISO8601(int32_t offset, UBool isBasic, UBool useUtcIndicator,
1473 UBool isShort, UBool ignoreSeconds, UnicodeString& result, UErrorCode& status) const {
1474 if (U_FAILURE(status)) {
1475 result.setToBogus();
1476 return result;
1477 }
1478 int32_t absOffset = offset < 0 ? -offset : offset;
1479 if (useUtcIndicator && (absOffset < MILLIS_PER_SECOND || (ignoreSeconds && absOffset < MILLIS_PER_MINUTE))) {
1480 result.setTo(ISO8601_UTC);
1481 return result;
1482 }
1483
1484 OffsetFields minFields = isShort ? FIELDS_H : FIELDS_HM;
1485 OffsetFields maxFields = ignoreSeconds ? FIELDS_HM : FIELDS_HMS;
1486 UChar sep = isBasic ? 0 : ISO8601_SEP;
1487
1488 // Note: FIELDS_HMS as maxFields is a CLDR/ICU extension. ISO 8601 specification does
1489 // not support seconds field.
1490
1491 if (absOffset >= MAX_OFFSET) {
1492 result.setToBogus();
1493 status = U_ILLEGAL_ARGUMENT_ERROR;
1494 return result;
1495 }
1496
1497 int fields[3];
1498 fields[0] = absOffset / MILLIS_PER_HOUR;
1499 absOffset = absOffset % MILLIS_PER_HOUR;
1500 fields[1] = absOffset / MILLIS_PER_MINUTE;
1501 absOffset = absOffset % MILLIS_PER_MINUTE;
1502 fields[2] = absOffset / MILLIS_PER_SECOND;
1503
1504 U_ASSERT(fields[0] >= 0 && fields[0] <= MAX_OFFSET_HOUR);
1505 U_ASSERT(fields[1] >= 0 && fields[1] <= MAX_OFFSET_MINUTE);
1506 U_ASSERT(fields[2] >= 0 && fields[2] <= MAX_OFFSET_SECOND);
1507
1508 int32_t lastIdx = maxFields;
1509 while (lastIdx > minFields) {
1510 if (fields[lastIdx] != 0) {
1511 break;
1512 }
1513 lastIdx--;
1514 }
1515
1516 UChar sign = PLUS;
1517 if (offset < 0) {
1518 // if all output fields are 0s, do not use negative sign
1519 for (int32_t idx = 0; idx <= lastIdx; idx++) {
1520 if (fields[idx] != 0) {
1521 sign = MINUS;
1522 break;
1523 }
1524 }
1525 }
1526 result.setTo(sign);
1527
1528 for (int32_t idx = 0; idx <= lastIdx; idx++) {
1529 if (sep && idx != 0) {
1530 result.append(sep);
1531 }
1532 result.append((UChar)(0x0030 + fields[idx]/10));
1533 result.append((UChar)(0x0030 + fields[idx]%10));
1534 }
1535
1536 return result;
1537}

1539UnicodeString&
1540TimeZoneFormat::formatOffsetLocalizedGMT(int32_t offset, UBool isShort, UnicodeString& result, UErrorCode& status) const {
1541 if (U_FAILURE(status)) {
1542 result.setToBogus();
1543 return result;
1544 }
1545 if (offset <= -MAX_OFFSET || offset >= MAX_OFFSET) {
1546 result.setToBogus();
1547 status = U_ILLEGAL_ARGUMENT_ERROR;
1548 return result;
1549 }
1550
1551 if (offset == 0) {
1552 result.setTo(fGMTZeroFormat);
1553 return result;
1554 }
1555
1556 UBool positive = TRUE1;
1557 if (offset < 0) {
1558 offset = -offset;
1559 positive = FALSE0;
1560 }
1561
1562 int32_t offsetH = offset / MILLIS_PER_HOUR;
1563 offset = offset % MILLIS_PER_HOUR;
1564 int32_t offsetM = offset / MILLIS_PER_MINUTE;
1565 offset = offset % MILLIS_PER_MINUTE;
1566 int32_t offsetS = offset / MILLIS_PER_SECOND;
1567
1568 U_ASSERT(offsetH <= MAX_OFFSET_HOUR && offsetM <= MAX_OFFSET_MINUTE && offsetS <= MAX_OFFSET_SECOND);
1569
1570 const UVector* offsetPatternItems = NULL__null;
1571 if (positive) {
1572 if (offsetS != 0) {
1573 offsetPatternItems = fGMTOffsetPatternItems[UTZFMT_PAT_POSITIVE_HMS];
1574 } else if (offsetM != 0 || !isShort) {
1575 offsetPatternItems = fGMTOffsetPatternItems[UTZFMT_PAT_POSITIVE_HM];
1576 } else {
1577 offsetPatternItems = fGMTOffsetPatternItems[UTZFMT_PAT_POSITIVE_H];
1578 }
1579 } else {
1580 if (offsetS != 0) {
1581 offsetPatternItems = fGMTOffsetPatternItems[UTZFMT_PAT_NEGATIVE_HMS];
1582 } else if (offsetM != 0 || !isShort) {
1583 offsetPatternItems = fGMTOffsetPatternItems[UTZFMT_PAT_NEGATIVE_HM];
1584 } else {
1585 offsetPatternItems = fGMTOffsetPatternItems[UTZFMT_PAT_NEGATIVE_H];
1586 }
1587 }
1588
1589 U_ASSERT(offsetPatternItems != NULL);
1590
1591 // Building the GMT format string
1592 result.setTo(fGMTPatternPrefix);
1593
1594 for (int32_t i = 0; i < offsetPatternItems->size(); i++) {
1595 const GMTOffsetField* item = (GMTOffsetField*)offsetPatternItems->elementAt(i);
1596 GMTOffsetField::FieldType type = item->getType();
1597
1598 switch (type) {
1599 case GMTOffsetField::TEXT:
1600 result.append(item->getPatternText(), -1);
1601 break;
1602
1603 case GMTOffsetField::HOUR:
1604 appendOffsetDigits(result, offsetH, (isShort ? 1 : 2));
1605 break;
1606
1607 case GMTOffsetField::MINUTE:
1608 appendOffsetDigits(result, offsetM, 2);
1609 break;
1610
1611 case GMTOffsetField::SECOND:
1612 appendOffsetDigits(result, offsetS, 2);
1613 break;
1614 }
1615 }
1616
1617 result.append(fGMTPatternSuffix);
1618 return result;
1619}

1621int32_t
1622TimeZoneFormat::parseOffsetISO8601(const UnicodeString& text, ParsePosition& pos, UBool extendedOnly, UBool* hasDigitOffset /* = NULL */) const {
1623 if (hasDigitOffset) {
1624 *hasDigitOffset = FALSE0;
1625 }
1626 int32_t start = pos.getIndex();
1627 if (start >= text.length()) {
1628 pos.setErrorIndex(start);
1629 return 0;
1630 }
1631
1632 UChar firstChar = text.charAt(start);
1633 if (firstChar == ISO8601_UTC || firstChar == (UChar)(ISO8601_UTC + 0x20)) {
1634 // "Z" (or "z") - indicates UTC
1635 pos.setIndex(start + 1);
1636 return 0;
1637 }
1638
1639 int32_t sign = 1;
1640 if (firstChar == PLUS) {
1641 sign = 1;
1642 } else if (firstChar == MINUS) {
1643 sign = -1;
1644 } else {
1645 // Not an ISO 8601 offset string
1646 pos.setErrorIndex(start);
1647 return 0;
1648 }
1649 ParsePosition posOffset(start + 1);
1650 int32_t offset = parseAsciiOffsetFields(text, posOffset, ISO8601_SEP, FIELDS_H, FIELDS_HMS);
1651 if (posOffset.getErrorIndex() == -1 && !extendedOnly && (posOffset.getIndex() - start <= 3)) {
1652 // If the text is successfully parsed as extended format with the options above, it can be also parsed
1653 // as basic format. For example, "0230" can be parsed as offset 2:00 (only first digits are valid for
1654 // extended format), but it can be parsed as offset 2:30 with basic format. We use longer result.
1655 ParsePosition posBasic(start + 1);
1656 int32_t tmpOffset = parseAbuttingAsciiOffsetFields(text, posBasic, FIELDS_H, FIELDS_HMS, FALSE0);
1657 if (posBasic.getErrorIndex() == -1 && posBasic.getIndex() > posOffset.getIndex()) {
1658 offset = tmpOffset;
1659 posOffset.setIndex(posBasic.getIndex());
1660 }
1661 }
1662
1663 if (posOffset.getErrorIndex() != -1) {
1664 pos.setErrorIndex(start);
1665 return 0;
1666 }
1667
1668 pos.setIndex(posOffset.getIndex());
1669 if (hasDigitOffset) {
1670 *hasDigitOffset = TRUE1;
1671 }
1672 return sign * offset;
1673}

1736int32_t
1737TimeZoneFormat::parseOffsetLocalizedGMTPattern(const UnicodeString& text, int32_t start, UBool /*isShort*/, int32_t& parsedLen) const {
1738 int32_t idx = start;
1739 int32_t offset = 0;
1740 UBool parsed = FALSE0;
1741
1742 do {
1743 // Prefix part
1744 int32_t len = fGMTPatternPrefix.length();
1745 if (len > 0 && text.caseCompare(idx, len, fGMTPatternPrefix, 0) != 0) {
1746 // prefix match failed
1747 break;
1748 }
1749 idx += len;
1750
1751 // Offset part
1752 offset = parseOffsetFields(text, idx, FALSE0, len);
1753 if (len == 0) {
1754 // offset field match failed
1755 break;
1756 }
1757 idx += len;
1758
1759 len = fGMTPatternSuffix.length();
1760 if (len > 0 && text.caseCompare(idx, len, fGMTPatternSuffix, 0) != 0) {
1761 // no suffix match
1762 break;
1763 }
1764 idx += len;
1765 parsed = TRUE1;
1766 } while (FALSE0);
1767
1768 parsedLen = parsed ? idx - start : 0;
1769 return offset;
1770}

1772int32_t
1773TimeZoneFormat::parseOffsetFields(const UnicodeString& text, int32_t start, UBool /*isShort*/, int32_t& parsedLen) const {
1774 int32_t outLen = 0;
1775 int32_t offset = 0;
1776 int32_t sign = 1;
1777
1778 parsedLen = 0;
1779
1780 int32_t offsetH, offsetM, offsetS;
1781 offsetH = offsetM = offsetS = 0;
1782
1783 for (int32_t patidx = 0; PARSE_GMT_OFFSET_TYPES[patidx] >= 0; patidx++) {
1784 int32_t gmtPatType = PARSE_GMT_OFFSET_TYPES[patidx];
1785 UVector* items = fGMTOffsetPatternItems[gmtPatType];
1786 U_ASSERT(items != NULL);
1787
1788 outLen = parseOffsetFieldsWithPattern(text, start, items, FALSE0, offsetH, offsetM, offsetS);
1789 if (outLen > 0) {
1790 sign = (gmtPatType == UTZFMT_PAT_POSITIVE_H || gmtPatType == UTZFMT_PAT_POSITIVE_HM || gmtPatType == UTZFMT_PAT_POSITIVE_HMS) ?
1791 1 : -1;
1792 break;
1793 }
1794 }
1795
1796 if (outLen > 0 && fAbuttingOffsetHoursAndMinutes) {
1797 // When hours field is sabutting minutes field,
1798 // the parse result above may not be appropriate.
1799 // For example, "01020" is parsed as 01:02: above,
1800 // but it should be parsed as 00:10:20.
1801 int32_t tmpLen = 0;
1802 int32_t tmpSign = 1;
1803 int32_t tmpH, tmpM, tmpS;
1804
1805 for (int32_t patidx = 0; PARSE_GMT_OFFSET_TYPES[patidx] >= 0; patidx++) {
1806 int32_t gmtPatType = PARSE_GMT_OFFSET_TYPES[patidx];
1807 UVector* items = fGMTOffsetPatternItems[gmtPatType];
1808 U_ASSERT(items != NULL);
1809
1810 // forcing parse to use single hour digit
1811 tmpLen = parseOffsetFieldsWithPattern(text, start, items, TRUE1, tmpH, tmpM, tmpS);
1812 if (tmpLen > 0) {
1813 tmpSign = (gmtPatType == UTZFMT_PAT_POSITIVE_H || gmtPatType == UTZFMT_PAT_POSITIVE_HM || gmtPatType == UTZFMT_PAT_POSITIVE_HMS) ?
1814 1 : -1;
1815 break;
1816 }
1817 }
1818 if (tmpLen > outLen) {
1819 // Better parse result with single hour digit
1820 outLen = tmpLen;
1821 sign = tmpSign;
1822 offsetH = tmpH;
1823 offsetM = tmpM;
1824 offsetS = tmpS;
1825 }
1826 }
1827
1828 if (outLen > 0) {
1829 offset = ((((offsetH * 60) + offsetM) * 60) + offsetS) * 1000 * sign;
1830 parsedLen = outLen;
1831 }
1832
1833 return offset;
1834}

1836int32_t
1837TimeZoneFormat::parseOffsetFieldsWithPattern(const UnicodeString& text, int32_t start,
1838 UVector* patternItems, UBool forceSingleHourDigit, int32_t& hour, int32_t& min, int32_t& sec) const {
1839 UBool failed = FALSE0;
1840 int32_t offsetH, offsetM, offsetS;
1841 offsetH = offsetM = offsetS = 0;
1842 int32_t idx = start;
1843
1844 for (int32_t i = 0; i < patternItems->size(); i++) {
1845 int32_t len = 0;
1846 const GMTOffsetField* field = (const GMTOffsetField*)patternItems->elementAt(i);
1847 GMTOffsetField::FieldType fieldType = field->getType();
