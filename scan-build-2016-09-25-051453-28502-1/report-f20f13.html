121// All the work is actually done here.
122//
123void RegexCompile::compile(
124 UText *pat, // Source pat to be compiled.
125 UParseError &pp, // Error position info
126 UErrorCode &e) // Error Code
127{
128 fStatus = &e;
129 fParseErr = &pp;
130 fStackPtr = 0;
131 fStack[fStackPtr] = 0;
132
133 if (U_FAILURE(*fStatus)) {
134 return;
135 }
136
137 // There should be no pattern stuff in the RegexPattern object. They can not be reused.
138 U_ASSERT(fRXPat->fPattern == NULL || utext_nativeLength(fRXPat->fPattern) == 0);
139
140 // Prepare the RegexPattern object to receive the compiled pattern.
141 fRXPat->fPattern = utext_cloneutext_clone_54(fRXPat->fPattern, pat, FALSE0, TRUE1, fStatus);
142 fRXPat->fStaticSets = RegexStaticSets::gStaticSets->fPropSets;
143 fRXPat->fStaticSets8 = RegexStaticSets::gStaticSets->fPropSets8;
144
145
146 // Initialize the pattern scanning state machine
147 fPatternLength = utext_nativeLengthutext_nativeLength_54(pat);
148 uint16_t state = 1;
149 const RegexTableEl *tableEl;
150
151 // UREGEX_LITERAL force entire pattern to be treated as a literal string.
152 if (fModeFlags & UREGEX_LITERAL) {
153 fQuoteMode = TRUE1;
154 }
155
156 nextChar(fC); // Fetch the first char from the pattern string.
157
158 //
159 // Main loop for the regex pattern parsing state machine.
160 // Runs once per state transition.
161 // Each time through optionally performs, depending on the state table,
162 // - an advance to the the next pattern char
163 // - an action to be performed.
164 // - pushing or popping a state to/from the local state return stack.
165 // file regexcst.txt is the source for the state table. The logic behind
166 // recongizing the pattern syntax is there, not here.
167 //
168 for (;;) {
169 // Bail out if anything has gone wrong.
170 // Regex pattern parsing stops on the first error encountered.
171 if (U_FAILURE(*fStatus)) {
172 break;
173 }
174
175 U_ASSERT(state != 0);
176
177 // Find the state table element that matches the input char from the pattern, or the
178 // class of the input character. Start with the first table row for this
179 // state, then linearly scan forward until we find a row that matches the
180 // character. The last row for each state always matches all characters, so
181 // the search will stop there, if not before.
182 //
183 tableEl = &gRuleParseStateTable[state];
184 REGEX_SCAN_DEBUG_PRINTF(("char, line, col = (\'%c\', %d, %d) state=%s ",
185 fC.fChar, fLineNum, fCharNum, RegexStateNames[state]));
186
187 for (;;) { // loop through table rows belonging to this state, looking for one
188 // that matches the current input char.
189 REGEX_SCAN_DEBUG_PRINTF(("."));
190 if (tableEl->fCharClass < 127 && fC.fQuoted == FALSE0 && tableEl->fCharClass == fC.fChar) {
191 // Table row specified an individual character, not a set, and
192 // the input character is not quoted, and
193 // the input character matched it.
194 break;
195 }
196 if (tableEl->fCharClass == 255) {
197 // Table row specified default, match anything character class.
198 break;
199 }
200 if (tableEl->fCharClass == 254 && fC.fQuoted) {
201 // Table row specified "quoted" and the char was quoted.
202 break;
203 }
204 if (tableEl->fCharClass == 253 && fC.fChar == (UChar32)-1) {
205 // Table row specified eof and we hit eof on the input.
206 break;
207 }
208
209 if (tableEl->fCharClass >= 128 && tableEl->fCharClass < 240 && // Table specs a char class &&
210 fC.fQuoted == FALSE0 && // char is not escaped &&
211 fC.fChar != (UChar32)-1) { // char is not EOF
212 U_ASSERT(tableEl->fCharClass <= 137);
213 if (RegexStaticSets::gStaticSets->fRuleSets[tableEl->fCharClass-128].contains(fC.fChar)) {
214 // Table row specified a character class, or set of characters,
215 // and the current char matches it.
216 break;
217 }
218 }
219
220 // No match on this row, advance to the next row for this state,
221 tableEl++;
222 }
223 REGEX_SCAN_DEBUG_PRINTF(("\n"));
224
225 //
226 // We've found the row of the state table that matches the current input
227 // character from the rules string.
228 // Perform any action specified by this row in the state table.
229 if (doParseActions(tableEl->fAction) == FALSE0) {
230 // Break out of the state machine loop if the
231 // the action signalled some kind of error, or
232 // the action was to exit, occurs on normal end-of-rules-input.
233 break;
234 }
235
236 if (tableEl->fPushState != 0) {
237 fStackPtr++;
238 if (fStackPtr >= kStackSize) {
239 error(U_REGEX_INTERNAL_ERROR);
240 REGEX_SCAN_DEBUG_PRINTF(("RegexCompile::parse() - state stack overflow.\n"));
241 fStackPtr--;
242 }
243 fStack[fStackPtr] = tableEl->fPushState;
244 }
245
246 //
247 // NextChar. This is where characters are actually fetched from the pattern.
248 // Happens under control of the 'n' tag in the state table.
249 //
250 if (tableEl->fNextChar) {
251 nextChar(fC);
252 }
253
254 // Get the next state from the table entry, or from the
255 // state stack if the next state was specified as "pop".
256 if (tableEl->fNextState != 255) {
257 state = tableEl->fNextState;
258 } else {
259 state = fStack[fStackPtr];
260 fStackPtr--;
261 if (fStackPtr < 0) {
262 // state stack underflow
263 // This will occur if the user pattern has mis-matched parentheses,
264 // with extra close parens.
265 //
266 fStackPtr++;
267 error(U_REGEX_MISMATCHED_PAREN);
268 }
269 }
270
271 }
272
273 if (U_FAILURE(*fStatus)) {
274 // Bail out if the pattern had errors.
275 // Set stack cleanup: a successful compile would have left it empty,
276 // but errors can leave temporary sets hanging around.
277 while (!fSetStack.empty()) {
278 delete (UnicodeSet *)fSetStack.pop();
279 }
280 return;
281 }
282
283 //
284 // The pattern has now been read and processed, and the compiled code generated.
285 //
286
287 //
288 // Compute the number of digits requried for the largest capture group number.
289 //
290 fRXPat->fMaxCaptureDigits = 1;
291 int32_t n = 10;
292 int32_t groupCount = fRXPat->fGroupMap->size();
293 while (n <= groupCount) {
294 fRXPat->fMaxCaptureDigits++;
295 n *= 10;
296 }
297
298 //
299 // The pattern's fFrameSize so far has accumulated the requirements for
300 // storage for capture parentheses, counters, etc. that are encountered
301 // in the pattern. Add space for the two variables that are always
302 // present in the saved state: the input string position (int64_t) and
303 // the position in the compiled pattern.
304 //
305 allocateStackData(RESTACKFRAME_HDRCOUNT2);
306
307 //
308 // Optimization pass 1: NOPs, back-references, and case-folding
309 //
310 stripNOPs();
311
312 //
313 // Get bounds for the minimum and maximum length of a string that this
314 // pattern can match. Used to avoid looking for matches in strings that
315 // are too short.
316 //
317 fRXPat->fMinMatchLen = minMatchLength(3, fRXPat->fCompiledPat->size()-1);
318
319 //
320 // Optimization pass 2: match start type
321 //
322 matchStartType();
323
324 //
325 // Set up fast latin-1 range sets
326 //
327 int32_t numSets = fRXPat->fSets->size();
328 fRXPat->fSets8 = new Regex8BitSet[numSets];
329 // Null pointer check.
330 if (fRXPat->fSets8 == NULL__null) {
331 e = *fStatus = U_MEMORY_ALLOCATION_ERROR;
332 return;
333 }
334 int32_t i;
335 for (i=0; i<numSets; i++) {
336 UnicodeSet *s = (UnicodeSet *)fRXPat->fSets->elementAt(i);
337 fRXPat->fSets8[i].init(s);
338 }
339
340}

346//------------------------------------------------------------------------------
347//
348// doParseAction Do some action during regex pattern parsing.
349// Called by the parse state machine.
350//
351// Generation of the match engine PCode happens here, or
352// in functions called from the parse actions defined here.
353//
354//
355//------------------------------------------------------------------------------
356UBool RegexCompile::doParseActions(int32_t action)
357{
358 UBool returnVal = TRUE1;
359
360 switch ((Regex_PatternParseAction)action) {
361
362 case doPatStart:
363 // Start of pattern compiles to:
364 //0 SAVE 2 Fall back to position of FAIL
365 //1 jmp 3
366 //2 FAIL Stop if we ever reach here.
367 //3 NOP Dummy, so start of pattern looks the same as
368 // the start of an ( grouping.
369 //4 NOP Resreved, will be replaced by a save if there are
370 // OR | operators at the top level
371 appendOp(URX_STATE_SAVE, 2);
372 appendOp(URX_JMP, 3);
373 appendOp(URX_FAIL, 0);
374
375 // Standard open nonCapture paren action emits the two NOPs and
376 // sets up the paren stack frame.
377 doParseActions(doOpenNonCaptureParen);
378 break;
379
380 case doPatFinish:
381 // We've scanned to the end of the pattern
382 // The end of pattern compiles to:
383 // URX_END
384 // which will stop the runtime match engine.
385 // Encountering end of pattern also behaves like a close paren,
386 // and forces fixups of the State Save at the beginning of the compiled pattern
387 // and of any OR operations at the top level.
388 //
389 handleCloseParen();
390 if (fParenStack.size() > 0) {
391 // Missing close paren in pattern.
392 error(U_REGEX_MISMATCHED_PAREN);
393 }
394
395 // add the END operation to the compiled pattern.
396 appendOp(URX_END, 0);
397
398 // Terminate the pattern compilation state machine.
399 returnVal = FALSE0;
400 break;
401
402
403
404 case doOrOperator:
405 // Scanning a '|', as in (A|B)
406 {
407 // Generate code for any pending literals preceding the '|'
408 fixLiterals(FALSE0);
409
410 // Insert a SAVE operation at the start of the pattern section preceding
411 // this OR at this level. This SAVE will branch the match forward
412 // to the right hand side of the OR in the event that the left hand
413 // side fails to match and backtracks. Locate the position for the
414 // save from the location on the top of the parentheses stack.
415 int32_t savePosition = fParenStack.popi();
416 int32_t op = (int32_t)fRXPat->fCompiledPat->elementAti(savePosition);
417 U_ASSERT(URX_TYPE(op) == URX_NOP); // original contents of reserved location
418 op = buildOp(URX_STATE_SAVE, fRXPat->fCompiledPat->size()+1);
419 fRXPat->fCompiledPat->setElementAt(op, savePosition);
420
421 // Append an JMP operation into the compiled pattern. The operand for
422 // the JMP will eventually be the location following the ')' for the
423 // group. This will be patched in later, when the ')' is encountered.
424 appendOp(URX_JMP, 0);
425
426 // Push the position of the newly added JMP op onto the parentheses stack.
427 // This registers if for fixup when this block's close paren is encountered.
428 fParenStack.push(fRXPat->fCompiledPat->size()-1, *fStatus);
429
430 // Append a NOP to the compiled pattern. This is the slot reserved
431 // for a SAVE in the event that there is yet another '|' following
432 // this one.
433 appendOp(URX_NOP, 0);
434 fParenStack.push(fRXPat->fCompiledPat->size()-1, *fStatus);
435 }
436 break;
437
438
439 case doOpenCaptureParen:
440 // Open Paren.
441 // Compile to a
442 // - NOP, which later may be replaced by a save-state if the
443 // parenthesized group gets a * quantifier, followed by
444 // - START_CAPTURE n where n is stack frame offset to the capture group variables.
445 // - NOP, which may later be replaced by a save-state if there
446 // is an '|' alternation within the parens.
447 //
448 // Each capture group gets three slots in the save stack frame:
449 // 0: Capture Group start position (in input string being matched.)
450 // 1: Capture Group end position.
451 // 2: Start of Match-in-progress.
452 // The first two locations are for a completed capture group, and are
453 // referred to by back references and the like.
454 // The third location stores the capture start position when an START_CAPTURE is
455 // encountered. This will be promoted to a completed capture when (and if) the corresponding
456 // END_CAPTURE is encountered.
457 {
458 fixLiterals();
459 appendOp(URX_NOP, 0);
460 int32_t varsLoc = allocateStackData(3); // Reserve three slots in match stack frame.
461 appendOp(URX_START_CAPTURE, varsLoc);
462 appendOp(URX_NOP, 0);
463
464 // On the Parentheses stack, start a new frame and add the postions
465 // of the two NOPs. Depending on what follows in the pattern, the
466 // NOPs may be changed to SAVE_STATE or JMP ops, with a target
467 // address of the end of the parenthesized group.
468 fParenStack.push(fModeFlags, *fStatus); // Match mode state
469 fParenStack.push(capturing, *fStatus); // Frame type.
470 fParenStack.push(fRXPat->fCompiledPat->size()-3, *fStatus); // The first NOP location
471 fParenStack.push(fRXPat->fCompiledPat->size()-1, *fStatus); // The second NOP loc
472
473 // Save the mapping from group number to stack frame variable position.
474 fRXPat->fGroupMap->addElement(varsLoc, *fStatus);
475 }
476 break;
477
478 case doOpenNonCaptureParen:
479 // Open non-caputuring (grouping only) Paren.
480 // Compile to a
481 // - NOP, which later may be replaced by a save-state if the
482 // parenthesized group gets a * quantifier, followed by
483 // - NOP, which may later be replaced by a save-state if there
484 // is an '|' alternation within the parens.
485 {
486 fixLiterals();
487 appendOp(URX_NOP, 0);
488 appendOp(URX_NOP, 0);
489
490 // On the Parentheses stack, start a new frame and add the postions
491 // of the two NOPs.
492 fParenStack.push(fModeFlags, *fStatus); // Match mode state
493 fParenStack.push(plain, *fStatus); // Begin a new frame.
494 fParenStack.push(fRXPat->fCompiledPat->size()-2, *fStatus); // The first NOP location
495 fParenStack.push(fRXPat->fCompiledPat->size()-1, *fStatus); // The second NOP loc
496 }
497 break;
498
499
500 case doOpenAtomicParen:
501 // Open Atomic Paren. (?>
502 // Compile to a
503 // - NOP, which later may be replaced if the parenthesized group
504 // has a quantifier, followed by
505 // - STO_SP save state stack position, so it can be restored at the ")"
506 // - NOP, which may later be replaced by a save-state if there
507 // is an '|' alternation within the parens.
508 {
509 fixLiterals();
510 appendOp(URX_NOP, 0);
511 int32_t varLoc = allocateData(1); // Reserve a data location for saving the state stack ptr.
512 appendOp(URX_STO_SP, varLoc);
513 appendOp(URX_NOP, 0);
514
515 // On the Parentheses stack, start a new frame and add the postions
516 // of the two NOPs. Depending on what follows in the pattern, the
517 // NOPs may be changed to SAVE_STATE or JMP ops, with a target
518 // address of the end of the parenthesized group.
519 fParenStack.push(fModeFlags, *fStatus); // Match mode state
520 fParenStack.push(atomic, *fStatus); // Frame type.
521 fParenStack.push(fRXPat->fCompiledPat->size()-3, *fStatus); // The first NOP
522 fParenStack.push(fRXPat->fCompiledPat->size()-1, *fStatus); // The second NOP
523 }
524 break;
525
526
527 case doOpenLookAhead:
528 // Positive Look-ahead (?= stuff )
529 //
530 // Note: Addition of transparent input regions, with the need to
531 // restore the original regions when failing out of a lookahead
532 // block, complicated this sequence. Some conbined opcodes
533 // might make sense - or might not, lookahead aren't that common.
534 //
535 // Caution: min match length optimization knows about this
536 // sequence; don't change without making updates there too.
537 //
538 // Compiles to
539 // 1 START_LA dataLoc Saves SP, Input Pos
540 // 2. STATE_SAVE 4 on failure of lookahead, goto 4
541 // 3 JMP 6 continue ...
542 //
543 // 4. LA_END Look Ahead failed. Restore regions.
544 // 5. BACKTRACK and back track again.
545 //
546 // 6. NOP reserved for use by quantifiers on the block.
547 // Look-ahead can't have quantifiers, but paren stack
548 // compile time conventions require the slot anyhow.
549 // 7. NOP may be replaced if there is are '|' ops in the block.
550 // 8. code for parenthesized stuff.
551 // 9. LA_END
552 //
553 // Two data slots are reserved, for saving the stack ptr and the input position.
554 {
555 fixLiterals();
556 int32_t dataLoc = allocateData(2);
557 appendOp(URX_LA_START, dataLoc);
558 appendOp(URX_STATE_SAVE, fRXPat->fCompiledPat->size()+ 2);
559 appendOp(URX_JMP, fRXPat->fCompiledPat->size()+ 3);
560 appendOp(URX_LA_END, dataLoc);
561 appendOp(URX_BACKTRACK, 0);
562 appendOp(URX_NOP, 0);
563 appendOp(URX_NOP, 0);
564
565 // On the Parentheses stack, start a new frame and add the postions
566 // of the NOPs.
567 fParenStack.push(fModeFlags, *fStatus); // Match mode state
568 fParenStack.push(lookAhead, *fStatus); // Frame type.
569 fParenStack.push(fRXPat->fCompiledPat->size()-2, *fStatus); // The first NOP location
570 fParenStack.push(fRXPat->fCompiledPat->size()-1, *fStatus); // The second NOP location
571 }
572 break;
573
574 case doOpenLookAheadNeg:
575 // Negated Lookahead. (?! stuff )
576 // Compiles to
577 // 1. START_LA dataloc
578 // 2. SAVE_STATE 7 // Fail within look-ahead block restores to this state,
579 // // which continues with the match.
580 // 3. NOP // Std. Open Paren sequence, for possible '|'
581 // 4. code for parenthesized stuff.
582 // 5. END_LA // Cut back stack, remove saved state from step 2.
583 // 6. BACKTRACK // code in block succeeded, so neg. lookahead fails.
584 // 7. END_LA // Restore match region, in case look-ahead was using
585 // an alternate (transparent) region.
586 {
587 fixLiterals();
588 int32_t dataLoc = allocateData(2);
589 appendOp(URX_LA_START, dataLoc);
590 appendOp(URX_STATE_SAVE, 0); // dest address will be patched later.
591 appendOp(URX_NOP, 0);
592
593 // On the Parentheses stack, start a new frame and add the postions
594 // of the StateSave and NOP.
595 fParenStack.push(fModeFlags, *fStatus); // Match mode state
596 fParenStack.push(negLookAhead, *fStatus); // Frame type
597 fParenStack.push(fRXPat->fCompiledPat->size()-2, *fStatus); // The STATE_SAVE location
598 fParenStack.push(fRXPat->fCompiledPat->size()-1, *fStatus); // The second NOP location
599
600 // Instructions #5 - #7 will be added when the ')' is encountered.
601 }
602 break;
603
604 case doOpenLookBehind:
605 {
606 // Compile a (?<= look-behind open paren.
607 //
608 // Compiles to
609 // 0 URX_LB_START dataLoc
610 // 1 URX_LB_CONT dataLoc
611 // 2 MinMatchLen
612 // 3 MaxMatchLen
613 // 4 URX_NOP Standard '(' boilerplate.
614 // 5 URX_NOP Reserved slot for use with '|' ops within (block).
615 // 6 <code for LookBehind expression>
616 // 7 URX_LB_END dataLoc # Check match len, restore input len
617 // 8 URX_LA_END dataLoc # Restore stack, input pos
618 //
619 // Allocate a block of matcher data, to contain (when running a match)
620 // 0: Stack ptr on entry
621 // 1: Input Index on entry
622 // 2: Start index of match current match attempt.
623 // 3: Original Input String len.
624
625 // Generate match code for any pending literals.
626 fixLiterals();
627
628 // Allocate data space
629 int32_t dataLoc = allocateData(4);
630
631 // Emit URX_LB_START
632 appendOp(URX_LB_START, dataLoc);
633
634 // Emit URX_LB_CONT
635 appendOp(URX_LB_CONT, dataLoc);
636 appendOp(URX_RESERVED_OP, 0); // MinMatchLength. To be filled later.
637 appendOp(URX_RESERVED_OP, 0); // MaxMatchLength. To be filled later.
638
639 // Emit the NOPs
640 appendOp(URX_NOP, 0);
641 appendOp(URX_NOP, 0);
642
643 // On the Parentheses stack, start a new frame and add the postions
644 // of the URX_LB_CONT and the NOP.
645 fParenStack.push(fModeFlags, *fStatus); // Match mode state
646 fParenStack.push(lookBehind, *fStatus); // Frame type
647 fParenStack.push(fRXPat->fCompiledPat->size()-2, *fStatus); // The first NOP location
648 fParenStack.push(fRXPat->fCompiledPat->size()-1, *fStatus); // The 2nd NOP location
649
650 // The final two instructions will be added when the ')' is encountered.
651 }
652
653 break;
654
655 case doOpenLookBehindNeg:
656 {
657 // Compile a (?<! negated look-behind open paren.
658 //
659 // Compiles to
660 // 0 URX_LB_START dataLoc # Save entry stack, input len
661 // 1 URX_LBN_CONT dataLoc # Iterate possible match positions
662 // 2 MinMatchLen
663 // 3 MaxMatchLen

1309 {

1311 int32_t topLoc = blockTopLoc(TRUE1);
1312 insertOp(topLoc);
1313
1314 // emit STO_SP loc
1315 int32_t stoLoc = allocateData(1); // Reserve the data location for storing save stack ptr.
1316 int32_t op = buildOp(URX_STO_SP, stoLoc);
1317 fRXPat->fCompiledPat->setElementAt(op, topLoc);
1318
1319 // Emit the SAVE_STATE 5
1320 int32_t L7 = fRXPat->fCompiledPat->size()+1;
1321 op = buildOp(URX_STATE_SAVE, L7);
1322 fRXPat->fCompiledPat->setElementAt(op, topLoc+1);
1323
1324 // Append the JMP operation.
1325 appendOp(URX_JMP, topLoc+1);
1326
1327 // Emit the LD_SP loc
1328 appendOp(URX_LD_SP, stoLoc);
1329 }
1330 break;
1331
1332 case doPossessiveOpt:
1333 // Possessive ?+ quantifier.
1334 // Compiles to
1335 // 1. STO_SP loc
1336 // 2. SAVE_STATE 5
1337 // 3. body of optional block
1338 // 4. LD_SP loc
1339 // 5. ...
1340 //
1341 {
1342 // Reserve two slots at the top of the block.
1343 int32_t topLoc = blockTopLoc(TRUE1);
1344 insertOp(topLoc);
1345
1346 // Emit the STO_SP
1347 int32_t stoLoc = allocateData(1); // Reserve the data location for storing save stack ptr.
1348 int32_t op = buildOp(URX_STO_SP, stoLoc);
1349 fRXPat->fCompiledPat->setElementAt(op, topLoc);
1350
1351 // Emit the SAVE_STATE
1352 int32_t continueLoc = fRXPat->fCompiledPat->size()+1;
1353 op = buildOp(URX_STATE_SAVE, continueLoc);
1354 fRXPat->fCompiledPat->setElementAt(op, topLoc+1);
1355
1356 // Emit the LD_SP
1357 appendOp(URX_LD_SP, stoLoc);
1358 }
1359 break;
1360
1361
1362 case doBeginMatchMode:
1363 fNewModeFlags = fModeFlags;
1364 fSetModeFlag = TRUE1;
1365 break;
1366
1367 case doMatchMode: // (?i) and similar
1368 {
1369 int32_t bit = 0;
1370 switch (fC.fChar) {
1371 case 0x69: /* 'i' */ bit = UREGEX_CASE_INSENSITIVE; break;
1372 case 0x64: /* 'd' */ bit = UREGEX_UNIX_LINES; break;
1373 case 0x6d: /* 'm' */ bit = UREGEX_MULTILINE; break;
1374 case 0x73: /* 's' */ bit = UREGEX_DOTALL; break;
1375 case 0x75: /* 'u' */ bit = 0; /* Unicode casing */ break;
1376 case 0x77: /* 'w' */ bit = UREGEX_UWORD; break;
1377 case 0x78: /* 'x' */ bit = UREGEX_COMMENTS; break;
1378 case 0x2d: /* '-' */ fSetModeFlag = FALSE0; break;
1379 default:
1380 U_ASSERT(FALSE); // Should never happen. Other chars are filtered out
1381 // by the scanner.
1382 }
1383 if (fSetModeFlag) {
1384 fNewModeFlags |= bit;
1385 } else {
1386 fNewModeFlags &= ~bit;
1387 }
1388 }
1389 break;
1390
1391 case doSetMatchMode:
1392 // Emit code to match any pending literals, using the not-yet changed match mode.
1393 fixLiterals();
1394
1395 // We've got a (?i) or similar. The match mode is being changed, but
1396 // the change is not scoped to a parenthesized block.
1397 U_ASSERT(fNewModeFlags < 0);
1398 fModeFlags = fNewModeFlags;
1399
1400 break;
1401
1402
1403 case doMatchModeParen:
1404 // We've got a (?i: or similar. Begin a parenthesized block, save old
1405 // mode flags so they can be restored at the close of the block.
1406 //
1407 // Compile to a
1408 // - NOP, which later may be replaced by a save-state if the
1409 // parenthesized group gets a * quantifier, followed by
1410 // - NOP, which may later be replaced by a save-state if there
1411 // is an '|' alternation within the parens.
1412 {
1413 fixLiterals(FALSE0);
1414 appendOp(URX_NOP, 0);
1415 appendOp(URX_NOP, 0);
1416
1417 // On the Parentheses stack, start a new frame and add the postions
1418 // of the two NOPs (a normal non-capturing () frame, except for the
1419 // saving of the orignal mode flags.)
1420 fParenStack.push(fModeFlags, *fStatus);
1421 fParenStack.push(flags, *fStatus); // Frame Marker
1422 fParenStack.push(fRXPat->fCompiledPat->size()-2, *fStatus); // The first NOP
1423 fParenStack.push(fRXPat->fCompiledPat->size()-1, *fStatus); // The second NOP
1424
1425 // Set the current mode flags to the new values.
1426 U_ASSERT(fNewModeFlags < 0);
1427 fModeFlags = fNewModeFlags;
1428 }
1429 break;
1430
1431 case doBadModeFlag:
1432 error(U_REGEX_INVALID_FLAG);
1433 break;
1434
1435 case doSuppressComments:
1436 // We have just scanned a '(?'. We now need to prevent the character scanner from
1437 // treating a '#' as a to-the-end-of-line comment.
1438 // (This Perl compatibility just gets uglier and uglier to do...)
1439 fEOLComments = FALSE0;
1440 break;
1441
1442
1443 case doSetAddAmp:
1444 {
1445 UnicodeSet *set = (UnicodeSet *)fSetStack.peek();
1446 set->add(chAmp);
1447 }
1448 break;
1449
1450 case doSetAddDash:
1451 {
1452 UnicodeSet *set = (UnicodeSet *)fSetStack.peek();
1453 set->add(chDash);
1454 }
1455 break;
1456
1457 case doSetBackslash_s:
1458 {
1459 UnicodeSet *set = (UnicodeSet *)fSetStack.peek();
1460 set->addAll(*RegexStaticSets::gStaticSets->fPropSets[URX_ISSPACE_SET]);
1461 break;
1462 }
1463
1464 case doSetBackslash_S:
1465 {
1466 UnicodeSet *set = (UnicodeSet *)fSetStack.peek();
1467 UnicodeSet SSet(*RegexStaticSets::gStaticSets->fPropSets[URX_ISSPACE_SET]);
1468 SSet.complement();
1469 set->addAll(SSet);
1470 break;
1471 }
1472
1473 case doSetBackslash_d:
1474 {
1475 UnicodeSet *set = (UnicodeSet *)fSetStack.peek();
1476 // TODO - make a static set, ticket 6058.
1477 addCategory(set, U_GC_ND_MASK((uint32_t)1<<(U_DECIMAL_DIGIT_NUMBER)), *fStatus);
1478 break;
1479 }
1480
1481 case doSetBackslash_D:
1482 {
1483 UnicodeSet *set = (UnicodeSet *)fSetStack.peek();
1484 UnicodeSet digits;
1485 // TODO - make a static set, ticket 6058.
1486 digits.applyIntPropertyValue(UCHAR_GENERAL_CATEGORY_MASK, U_GC_ND_MASK((uint32_t)1<<(U_DECIMAL_DIGIT_NUMBER)), *fStatus);
1487 digits.complement();
1488 set->addAll(digits);
1489 break;
1490 }
1491
1492 case doSetBackslash_w:
1493 {
1494 UnicodeSet *set = (UnicodeSet *)fSetStack.peek();
1495 set->addAll(*RegexStaticSets::gStaticSets->fPropSets[URX_ISWORD_SET]);
1496 break;
1497 }
1498
1499 case doSetBackslash_W:
1500 {
1501 UnicodeSet *set = (UnicodeSet *)fSetStack.peek();
1502 UnicodeSet SSet(*RegexStaticSets::gStaticSets->fPropSets[URX_ISWORD_SET]);
1503 SSet.complement();
1504 set->addAll(SSet);
1505 break;
1506 }
1507
1508 case doSetBegin:
1509 fixLiterals(FALSE0);
1510 fSetStack.push(new UnicodeSet(), *fStatus);
1511 fSetOpStack.push(setStart, *fStatus);
1512 if ((fModeFlags & UREGEX_CASE_INSENSITIVE) != 0) {
1513 fSetOpStack.push(setCaseClose, *fStatus);
1514 }
1515 break;
1516
1517 case doSetBeginDifference1:
1518 // We have scanned something like [[abc]-[
1519 // Set up a new UnicodeSet for the set beginning with the just-scanned '['
1520 // Push a Difference operator, which will cause the new set to be subtracted from what
1521 // went before once it is created.
1522 setPushOp(setDifference1);
1523 fSetOpStack.push(setStart, *fStatus);
1524 if ((fModeFlags & UREGEX_CASE_INSENSITIVE) != 0) {
1525 fSetOpStack.push(setCaseClose, *fStatus);
1526 }
1527 break;
1528
1529 case doSetBeginIntersection1:
1530 // We have scanned something like [[abc]&[
1531 // Need both the '&' operator and the open '[' operator.
1532 setPushOp(setIntersection1);
1533 fSetOpStack.push(setStart, *fStatus);
1534 if ((fModeFlags & UREGEX_CASE_INSENSITIVE) != 0) {
1535 fSetOpStack.push(setCaseClose, *fStatus);
1536 }
1537 break;
1538
1539 case doSetBeginUnion:
1540 // We have scanned something like [[abc][
1541 // Need to handle the union operation explicitly [[abc] | [
1542 setPushOp(setUnion);
1543 fSetOpStack.push(setStart, *fStatus);
1544 if ((fModeFlags & UREGEX_CASE_INSENSITIVE) != 0) {
1545 fSetOpStack.push(setCaseClose, *fStatus);
1546 }
1547 break;
1548
1549 case doSetDifference2:
1550 // We have scanned something like [abc--
1551 // Consider this to unambiguously be a set difference operator.
1552 setPushOp(setDifference2);
1553 break;
1554
1555 case doSetEnd:
1556 // Have encountered the ']' that closes a set.
1557 // Force the evaluation of any pending operations within this set,
1558 // leave the completed set on the top of the set stack.
1559 setEval(setEnd);
1560 U_ASSERT(fSetOpStack.peeki()==setStart);
1561 fSetOpStack.popi();
1562 break;
1563
1564 case doSetFinish:
1565 {
1566 // Finished a complete set expression, including all nested sets.
1567 // The close bracket has already triggered clearing out pending set operators,
1568 // the operator stack should be empty and the operand stack should have just
1569 // one entry, the result set.
1570 U_ASSERT(fSetOpStack.empty());
1571 UnicodeSet *theSet = (UnicodeSet *)fSetStack.pop();
1572 U_ASSERT(fSetStack.empty());
1573 compileSet(theSet);
1574 break;
1575 }
1576
1577 case doSetIntersection2:
1578 // Have scanned something like [abc&&
1579 setPushOp(setIntersection2);
1580 break;
1581
1582 case doSetLiteral:
1583 // Union the just-scanned literal character into the set being built.
1584 // This operation is the highest precedence set operation, so we can always do
1585 // it immediately, without waiting to see what follows. It is necessary to perform
1586 // any pending '-' or '&' operation first, because these have the same precedence
1587 // as union-ing in a literal'
1588 {
1589 setEval(setUnion);
1590 UnicodeSet *s = (UnicodeSet *)fSetStack.peek();
1591 s->add(fC.fChar);
1592 fLastSetLiteral = fC.fChar;
1593 break;
1594 }
1595
1596 case doSetLiteralEscaped:
1597 // A back-slash escaped literal character was encountered.
1598 // Processing is the same as with setLiteral, above, with the addition of
1599 // the optional check for errors on escaped ASCII letters.
1600 {
1601 if ((fModeFlags & UREGEX_ERROR_ON_UNKNOWN_ESCAPES) != 0 &&
1602 ((fC.fChar >= 0x41 && fC.fChar<= 0x5A) || // in [A-Z]
1603 (fC.fChar >= 0x61 && fC.fChar <= 0x7a))) { // in [a-z]
1604 error(U_REGEX_BAD_ESCAPE_SEQUENCE);
1605 }
1606 setEval(setUnion);
1607 UnicodeSet *s = (UnicodeSet *)fSetStack.peek();
1608 s->add(fC.fChar);
1609 fLastSetLiteral = fC.fChar;
1610 break;
1611 }
1612
1613 case doSetNamedChar:
1614 // Scanning a \N{UNICODE CHARACTER NAME}
1615 // Aside from the source of the character, the processing is identical to doSetLiteral,
1616 // above.
1617 {
1618 UChar32 c = scanNamedChar();
1619 setEval(setUnion);
1620 UnicodeSet *s = (UnicodeSet *)fSetStack.peek();
1621 s->add(c);
1622 fLastSetLiteral = c;
1623 break;
1624 }
1625
1626 case doSetNamedRange:
1627 // We have scanned literal-\N{CHAR NAME}. Add the range to the set.
1628 // The left character is already in the set, and is saved in fLastSetLiteral.
1629 // Nonetheless, check if |fLastSetLiteral| is indeed set because it's
1630 // not set in some edge cases.
1631 // The right side needs to be picked up, the scan is at the 'N'.
1632 // Lower Limit > Upper limit being an error matches both Java
1633 // and ICU UnicodeSet behavior.
1634 {
1635 UChar32 c = scanNamedChar();
1636 if (U_SUCCESS(*fStatus) && (fLastSetLiteral == U_SENTINEL(-1) || fLastSetLiteral > c)) {
1637 error(U_REGEX_INVALID_RANGE);
1638 }
1639 UnicodeSet *s = (UnicodeSet *)fSetStack.peek();
1640 s->add(fLastSetLiteral, c);
1641 fLastSetLiteral = c;
1642 break;
1643 }
1644
1645
1646 case doSetNegate:
1647 // Scanned a '^' at the start of a set.
1648 // Push the negation operator onto the set op stack.
1649 // A twist for case-insensitive matching:
1650 // the case closure operation must happen _before_ negation.
1651 // But the case closure operation will already be on the stack if it's required.
1652 // This requires checking for case closure, and swapping the stack order
1653 // if it is present.
1654 {
1655 int32_t tosOp = fSetOpStack.peeki();
1656 if (tosOp == setCaseClose) {
1657 fSetOpStack.popi();
1658 fSetOpStack.push(setNegation, *fStatus);
1659 fSetOpStack.push(setCaseClose, *fStatus);
1660 } else {
1661 fSetOpStack.push(setNegation, *fStatus);
1662 }
1663 }
1664 break;
1665
1666 case doSetNoCloseError:
1667 error(U_REGEX_MISSING_CLOSE_BRACKET);
1668 break;
1669
1670 case doSetOpError:
1671 error(U_REGEX_RULE_SYNTAX); // -- or && at the end of a set. Illegal.
1672 break;
1673
1674 case doSetPosixProp:
1675 {
1676 UnicodeSet *s = scanPosixProp();
1677 if (s != NULL__null) {
1678 UnicodeSet *tos = (UnicodeSet *)fSetStack.peek();
1679 tos->addAll(*s);
1680 delete s;
1681 } // else error. scanProp() reported the error status already.
1682 }
1683 break;
1684
1685 case doSetProp:
1686 // Scanned a \p \P within [brackets].
1687 {
1688 UnicodeSet *s = scanProp();
1689 if (s != NULL__null) {
1690 UnicodeSet *tos = (UnicodeSet *)fSetStack.peek();
1691 tos->addAll(*s);
1692 delete s;
1693 } // else error. scanProp() reported the error status already.
1694 }
1695 break;
1696
1697
1698 case doSetRange:
1699 // We have scanned literal-literal. Add the range to the set.
1700 // The left character is already in the set, and is saved in fLastSetLiteral.
1701 // Nonetheless, check if |fLastSetLiteral| is indeed set because it's
1702 // not set in some edge cases.
1703 // The right side is the current character.
1704 // Lower Limit > Upper limit being an error matches both Java
1705 // and ICU UnicodeSet behavior.
1706 {
1707 if (fLastSetLiteral == U_SENTINEL(-1) || fLastSetLiteral > fC.fChar) {
1708 error(U_REGEX_INVALID_RANGE);
1709 }
1710 UnicodeSet *s = (UnicodeSet *)fSetStack.peek();
1711 s->add(fLastSetLiteral, fC.fChar);
1712 break;
1713 }
1714
1715 default:
1716 U_ASSERT(FALSE);
1717 error(U_REGEX_INTERNAL_ERROR);
1718 break;
1719 }
1720
1721 if (U_FAILURE(*fStatus)) {
1722 returnVal = FALSE0;
1723 }
1724
1725 return returnVal;
1726}

1743//------------------------------------------------------------------------------
1744//
1745// fixLiterals When compiling something that can follow a literal
1746// string in a pattern, emit the code to match the
1747// accumulated literal string.
1748//
1749// Optionally, split the last char of the string off into
1750// a single "ONE_CHAR" operation, so that quantifiers can
1751// apply to that char alone. Example: abc*
1752// The * must apply to the 'c' only.
1753//
1754//------------------------------------------------------------------------------
1755void RegexCompile::fixLiterals(UBool split) {
1756
1757 // If no literal characters have been scanned but not yet had code generated
1758 // for them, nothing needs to be done.
1759 if (fLiteralChars.length() == 0) {
1760 return;
1761 }
1762
1763 int32_t indexOfLastCodePoint = fLiteralChars.moveIndex32(fLiteralChars.length(), -1);
1764 UChar32 lastCodePoint = fLiteralChars.char32At(indexOfLastCodePoint);
1765
1766 // Split: We need to ensure that the last item in the compiled pattern
1767 // refers only to the last literal scanned in the pattern, so that
1768 // quantifiers (*, +, etc.) affect only it, and not a longer string.
1769 // Split before case folding for case insensitive matches.
1770
1771 if (split) {
1772 fLiteralChars.truncate(indexOfLastCodePoint);
1773 fixLiterals(FALSE0); // Recursive call, emit code to match the first part of the string.
1774 // Note that the truncated literal string may be empty, in which case
1775 // nothing will be emitted.
1776
1777 literalChar(lastCodePoint); // Re-add the last code point as if it were a new literal.
1778 fixLiterals(FALSE0); // Second recursive call, code for the final code point.
1779 return;
1780 }
1781
1782 // If we are doing case-insensitive matching, case fold the string. This may expand
1783 // the string, e.g. the German sharp-s turns into "ss"
1784 if (fModeFlags & UREGEX_CASE_INSENSITIVE) {
1785 fLiteralChars.foldCase();
1786 indexOfLastCodePoint = fLiteralChars.moveIndex32(fLiteralChars.length(), -1);
1787 lastCodePoint = fLiteralChars.char32At(indexOfLastCodePoint);
1788 }
1789
1790 if (indexOfLastCodePoint == 0) {
1791 // Single character, emit a URX_ONECHAR op to match it.
1792 if ((fModeFlags & UREGEX_CASE_INSENSITIVE) &&
1793 u_hasBinaryPropertyu_hasBinaryProperty_54(lastCodePoint, UCHAR_CASE_SENSITIVE)) {
1794 appendOp(URX_ONECHAR_I, lastCodePoint);
1795 } else {
1796 appendOp(URX_ONECHAR, lastCodePoint);
1797 }
1798 } else {
1799 // Two or more chars, emit a URX_STRING to match them.
1800 if (fLiteralChars.length() > 0x00ffffff || fRXPat->fLiteralText.length() > 0x00ffffff) {
1801 error(U_REGEX_PATTERN_TOO_BIG);
1802 }
1803 if (fModeFlags & UREGEX_CASE_INSENSITIVE) {
1804 appendOp(URX_STRING_I, fRXPat->fLiteralText.length());
1805 } else {
1806 // TODO here: add optimization to split case sensitive strings of length two
1807 // into two single char ops, for efficiency.
1808 appendOp(URX_STRING, fRXPat->fLiteralText.length());
1809 }
1810 appendOp(URX_STRING_LEN, fLiteralChars.length());
1811
1812 // Add this string into the accumulated strings of the compiled pattern.
1813 fRXPat->fLiteralText.append(fLiteralChars);
1814 }
1815
1816 fLiteralChars.remove();
1817}

1820int32_t RegexCompile::buildOp(int32_t type, int32_t val) {
1821 if (U_FAILURE(*fStatus)) {
1822 return 0;
1823 }
1824 if (type < 0 || type > 255) {
1825 U_ASSERT(FALSE);
1826 error(U_REGEX_INTERNAL_ERROR);
1827 type = URX_RESERVED_OP;
1828 }
1829 if (val > 0x00ffffff) {
1830 U_ASSERT(FALSE);
1831 error(U_REGEX_INTERNAL_ERROR);
1832 val = 0;
1833 }
1834 if (val < 0) {
1835 if (!(type == URX_RESERVED_OP_N || type == URX_RESERVED_OP)) {
1836 U_ASSERT(FALSE);
1837 error(U_REGEX_INTERNAL_ERROR);
1838 return -1;
1839 }
1840 if (URX_TYPE(val)((uint32_t)(val) >> 24) != 0xff) {
1841 U_ASSERT(FALSE);
1842 error(U_REGEX_INTERNAL_ERROR);
1843 return -1;
1844 }
1845 type = URX_RESERVED_OP_N;
1846 }
1847 return (type << 24) | val;
1848}

1874//------------------------------------------------------------------------------
1875//
1876// insertOp() Insert a slot for a new opcode into the already
1877// compiled pattern code.
1878//
1879// Fill the slot with a NOP. Our caller will replace it
1880// with what they really wanted.
1881//
1882//------------------------------------------------------------------------------
1883void RegexCompile::insertOp(int32_t where) {
1884 UVector64 *code = fRXPat->fCompiledPat;
1885 U_ASSERT(where>0 && where < code->size());
1886
1887 int32_t nop = buildOp(URX_NOP, 0);
1888 code->insertElementAt(nop, where, *fStatus);
1889
1890 // Walk through the pattern, looking for any ops with targets that
1891 // were moved down by the insert. Fix them.
1892 int32_t loc;
1893 for (loc=0; loc<code->size(); loc++) {
1894 int32_t op = (int32_t)code->elementAti(loc);
1895 int32_t opType = URX_TYPE(op)((uint32_t)(op) >> 24);
1896 int32_t opValue = URX_VAL(op)((op) & 0xffffff);
1897 if ((opType == URX_JMP ||
1898 opType == URX_JMPX ||
1899 opType == URX_STATE_SAVE ||
1900 opType == URX_CTR_LOOP ||
1901 opType == URX_CTR_LOOP_NG ||
1902 opType == URX_JMP_SAV ||
1903 opType == URX_JMP_SAV_X ||
1904 opType == URX_RELOC_OPRND) && opValue > where) {
1905 // Target location for this opcode is after the insertion point and
1906 // needs to be incremented to adjust for the insertion.
1907 opValue++;
1908 op = buildOp(opType, opValue);
1909 code->setElementAt(op, loc);
1910 }
1911 }
1912
1913 // Now fix up the parentheses stack. All positive values in it are locations in
1914 // the compiled pattern. (Negative values are frame boundaries, and don't need fixing.)
1915 for (loc=0; loc<fParenStack.size(); loc++) {
1916 int32_t x = fParenStack.elementAti(loc);
1917 U_ASSERT(x < code->size());
1918 if (x>where) {
1919 x++;
1920 fParenStack.setElementAt(x, loc);
1921 }
1922 }
1923
1924 if (fMatchCloseParen > where) {
1925 fMatchCloseParen++;
1926 }
1927 if (fMatchOpenParen > where) {
1928 fMatchOpenParen++;
1929 }
1930}

1986//------------------------------------------------------------------------------
1987//
1988// blockTopLoc() Find or create a location in the compiled pattern
1989// at the start of the operation or block that has
1990// just been compiled. Needed when a quantifier (* or
1991// whatever) appears, and we need to add an operation
1992// at the start of the thing being quantified.
1993//
1994// (Parenthesized Blocks) have a slot with a NOP that
1995// is reserved for this purpose. .* or similar don't
1996// and a slot needs to be added.
1997//
1998// parameter reserveLoc : TRUE - ensure that there is space to add an opcode
1999// at the returned location.
2000// FALSE - just return the address,
2001// do not reserve a location there.
2002//
2003//------------------------------------------------------------------------------
2004int32_t RegexCompile::blockTopLoc(UBool reserveLoc) {
2005 int32_t theLoc;
2006 fixLiterals(TRUE1); // Emit code for any pending literals.
2007 // If last item was a string, emit separate op for the its last char.
2008 if (fRXPat->fCompiledPat->size() == fMatchCloseParen)
2009 {
2010 // The item just processed is a parenthesized block.
2011 theLoc = fMatchOpenParen; // A slot is already reserved for us.
2012 U_ASSERT(theLoc > 0);
2013 U_ASSERT(URX_TYPE(((uint32_t)fRXPat->fCompiledPat->elementAti(theLoc))) == URX_NOP);
2014 }
2015 else {
2016 // Item just compiled is a single thing, a ".", or a single char, a string or a set reference.
2017 // No slot for STATE_SAVE was pre-reserved in the compiled code.
2018 // We need to make space now.
2019 theLoc = fRXPat->fCompiledPat->size()-1;
2020 int32_t opAtTheLoc = (int32_t)fRXPat->fCompiledPat->elementAti(theLoc);
2021 if (URX_TYPE(opAtTheLoc)((uint32_t)(opAtTheLoc) >> 24) == URX_STRING_LEN) {
2022 // Strings take two opcode, we want the position of the first one.
2023 // We can have a string at this point if a single character case-folded to two.
2024 theLoc--;
2025 }
2026 if (reserveLoc) {
2027 int32_t nop = buildOp(URX_NOP, 0);
2028 fRXPat->fCompiledPat->insertElementAt(nop, theLoc, *fStatus);
2029 }
2030 }
2031 return theLoc;
2032}

2036//------------------------------------------------------------------------------
2037//
2038// handleCloseParen When compiling a close paren, we need to go back
2039// and fix up any JMP or SAVE operations within the
2040// parenthesized block that need to target the end
2041// of the block. The locations of these are kept on
2042// the paretheses stack.
2043//
2044// This function is called both when encountering a
2045// real ) and at the end of the pattern.
2046//
2047//------------------------------------------------------------------------------
2048void RegexCompile::handleCloseParen() {
2049 int32_t patIdx;
2050 int32_t patOp;
2051 if (fParenStack.size() <= 0) {
2052 error(U_REGEX_MISMATCHED_PAREN);
2053 return;
2054 }
2055
2056 // Emit code for any pending literals.
2057 fixLiterals(FALSE0);
2058
2059 // Fixup any operations within the just-closed parenthesized group
2060 // that need to reference the end of the (block).
2061 // (The first one popped from the stack is an unused slot for
2062 // alternation (OR) state save, but applying the fixup to it does no harm.)
2063 for (;;) {
2064 patIdx = fParenStack.popi();
2065 if (patIdx < 0) {
2066 // value < 0 flags the start of the frame on the paren stack.
2067 break;
2068 }
2069 U_ASSERT(patIdx>0 && patIdx <= fRXPat->fCompiledPat->size());
2070 patOp = (int32_t)fRXPat->fCompiledPat->elementAti(patIdx);
2071 U_ASSERT(URX_VAL(patOp) == 0); // Branch target for JMP should not be set.
2072 patOp |= fRXPat->fCompiledPat->size(); // Set it now.
2073 fRXPat->fCompiledPat->setElementAt(patOp, patIdx);
2074 fMatchOpenParen = patIdx;
2075 }
2076
2077 // At the close of any parenthesized block, restore the match mode flags to
2078 // the value they had at the open paren. Saved value is
2079 // at the top of the paren stack.
2080 fModeFlags = fParenStack.popi();
2081 U_ASSERT(fModeFlags < 0);
2082
2083 // DO any additional fixups, depending on the specific kind of
2084 // parentesized grouping this is
2085
2086 switch (patIdx) {
2087 case plain:
2088 case flags:
2089 // No additional fixups required.
2090 // (Grouping-only parentheses)
2091 break;
2092 case capturing:
2093 // Capturing Parentheses.
2094 // Insert a End Capture op into the pattern.
2095 // The frame offset of the variables for this cg is obtained from the
2096 // start capture op and put it into the end-capture op.
2097 {
2098 int32_t captureOp = (int32_t)fRXPat->fCompiledPat->elementAti(fMatchOpenParen+1);
2099 U_ASSERT(URX_TYPE(captureOp) == URX_START_CAPTURE);
2100
2101 int32_t frameVarLocation = URX_VAL(captureOp)((captureOp) & 0xffffff);
2102 appendOp(URX_END_CAPTURE, frameVarLocation);
2103 }
2104 break;
2105 case atomic:
2106 // Atomic Parenthesis.
2107 // Insert a LD_SP operation to restore the state stack to the position
2108 // it was when the atomic parens were entered.
2109 {
2110 int32_t stoOp = (int32_t)fRXPat->fCompiledPat->elementAti(fMatchOpenParen+1);
2111 U_ASSERT(URX_TYPE(stoOp) == URX_STO_SP);
2112 int32_t stoLoc = URX_VAL(stoOp)((stoOp) & 0xffffff);
2113 appendOp(URX_LD_SP, stoLoc);
2114 }
2115 break;
2116
2117 case lookAhead:
2118 {
2119 int32_t startOp = (int32_t)fRXPat->fCompiledPat->elementAti(fMatchOpenParen-5);
2120 U_ASSERT(URX_TYPE(startOp) == URX_LA_START);
2121 int32_t dataLoc = URX_VAL(startOp)((startOp) & 0xffffff);
2122 appendOp(URX_LA_END, dataLoc);
2123 }
2124 break;
2125
2126 case negLookAhead:
2127 {
2128 // See comment at doOpenLookAheadNeg
2129 int32_t startOp = (int32_t)fRXPat->fCompiledPat->elementAti(fMatchOpenParen-1);
2130 U_ASSERT(URX_TYPE(startOp) == URX_LA_START);
2131 int32_t dataLoc = URX_VAL(startOp)((startOp) & 0xffffff);
2132 appendOp(URX_LA_END, dataLoc);
2133 appendOp(URX_BACKTRACK, 0);
2134 appendOp(URX_LA_END, dataLoc);
2135
2136 // Patch the URX_SAVE near the top of the block.
2137 // The destination of the SAVE is the final LA_END that was just added.
2138 int32_t saveOp = (int32_t)fRXPat->fCompiledPat->elementAti(fMatchOpenParen);
2139 U_ASSERT(URX_TYPE(saveOp) == URX_STATE_SAVE);
2140 int32_t dest = fRXPat->fCompiledPat->size()-1;
2141 saveOp = buildOp(URX_STATE_SAVE, dest);
2142 fRXPat->fCompiledPat->setElementAt(saveOp, fMatchOpenParen);
2143 }
2144 break;
2145
2146 case lookBehind:
2147 {
2148 // See comment at doOpenLookBehind.
2149
2150 // Append the URX_LB_END and URX_LA_END to the compiled pattern.
2151 int32_t startOp = (int32_t)fRXPat->fCompiledPat->elementAti(fMatchOpenParen-4);
2152 U_ASSERT(URX_TYPE(startOp) == URX_LB_START);
2153 int32_t dataLoc = URX_VAL(startOp)((startOp) & 0xffffff);
2154 appendOp(URX_LB_END, dataLoc);
2155 appendOp(URX_LA_END, dataLoc);

2157 // Determine the min and max bounds for the length of the
2158 // string that the pattern can match.
2159 // An unbounded upper limit is an error.
2160 int32_t patEnd = fRXPat->fCompiledPat->size() - 1;
2161 int32_t minML = minMatchLength(fMatchOpenParen, patEnd);
2162 int32_t maxML = maxMatchLength(fMatchOpenParen, patEnd);
2163 if (URX_TYPE(maxML)((uint32_t)(maxML) >> 24) != 0) {
2164 error(U_REGEX_LOOK_BEHIND_LIMIT);
2165 break;
2166 }
2167 if (maxML == INT32_MAX(2147483647)) {
2168 error(U_REGEX_LOOK_BEHIND_LIMIT);
2169 break;
2170 }
2171 U_ASSERT(minML <= maxML);
2172
2173 // Insert the min and max match len bounds into the URX_LB_CONT op that
2174 // appears at the top of the look-behind block, at location fMatchOpenParen+1
2175 fRXPat->fCompiledPat->setElementAt(minML, fMatchOpenParen-2);
2176 fRXPat->fCompiledPat->setElementAt(maxML, fMatchOpenParen-1);
2177
2178 }
2179 break;
2180
2181
2182
2183 case lookBehindN:
2184 {
2185 // See comment at doOpenLookBehindNeg.
2186
2187 // Append the URX_LBN_END to the compiled pattern.
2188 int32_t startOp = (int32_t)fRXPat->fCompiledPat->elementAti(fMatchOpenParen-5);
2189 U_ASSERT(URX_TYPE(startOp) == URX_LB_START);
2190 int32_t dataLoc = URX_VAL(startOp)((startOp) & 0xffffff);
2191 appendOp(URX_LBN_END, dataLoc);
2192
2193 // Determine the min and max bounds for the length of the
2194 // string that the pattern can match.
2195 // An unbounded upper limit is an error.
2196 int32_t patEnd = fRXPat->fCompiledPat->size() - 1;
2197 int32_t minML = minMatchLength(fMatchOpenParen, patEnd);
2198 int32_t maxML = maxMatchLength(fMatchOpenParen, patEnd);
2199 if (URX_TYPE(maxML)((uint32_t)(maxML) >> 24) != 0) {
2200 error(U_REGEX_LOOK_BEHIND_LIMIT);
2201 break;
2202 }
2203 if (maxML == INT32_MAX(2147483647)) {
2204 error(U_REGEX_LOOK_BEHIND_LIMIT);
2205 break;
2206 }
2207 U_ASSERT(minML <= maxML);
2208
2209 // Insert the min and max match len bounds into the URX_LB_CONT op that
2210 // appears at the top of the look-behind block, at location fMatchOpenParen+1
2211 fRXPat->fCompiledPat->setElementAt(minML, fMatchOpenParen-3);
2212 fRXPat->fCompiledPat->setElementAt(maxML, fMatchOpenParen-2);
2213
2214 // Insert the pattern location to continue at after a successful match
2215 // as the last operand of the URX_LBN_CONT
2216 int32_t op = buildOp(URX_RELOC_OPRND, fRXPat->fCompiledPat->size());
2217 fRXPat->fCompiledPat->setElementAt(op, fMatchOpenParen-1);
2218 }
2219 break;
2220
2221
2222
2223 default:
2224 U_ASSERT(FALSE);
2225 }
2226
2227 // remember the next location in the compiled pattern.
2228 // The compilation of Quantifiers will look at this to see whether its looping
2229 // over a parenthesized block or a single item
2230 fMatchCloseParen = fRXPat->fCompiledPat->size();
2231}

2284//------------------------------------------------------------------------------
2285//
2286// compileInterval Generate the code for a {min, max} style interval quantifier.
2287// Except for the specific opcodes used, the code is the same
2288// for all three types (greedy, non-greedy, possessive) of
2289// intervals. The opcodes are supplied as parameters.
2290// (There are two sets of opcodes - greedy & possessive use the
2291// same ones, while non-greedy has it's own.)
2292//
2293// The code for interval loops has this form:
2294// 0 CTR_INIT counter loc (in stack frame)
2295// 1 5 patt address of CTR_LOOP at bottom of block
2296// 2 min count
2297// 3 max count (-1 for unbounded)
2298// 4 ... block to be iterated over
2299// 5 CTR_LOOP
2300//
2301// In
2302//------------------------------------------------------------------------------
2303void RegexCompile::compileInterval(int32_t InitOp, int32_t LoopOp)
2304{
2305 // The CTR_INIT op at the top of the block with the {n,m} quantifier takes
2306 // four slots in the compiled code. Reserve them.
2307 int32_t topOfBlock = blockTopLoc(TRUE1);
2308 insertOp(topOfBlock);
2309 insertOp(topOfBlock);
2310 insertOp(topOfBlock);
2311
2312 // The operands for the CTR_INIT opcode include the index in the matcher data
2313 // of the counter. Allocate it now. There are two data items
2314 // counterLoc --> Loop counter
2315 // +1 --> Input index (for breaking non-progressing loops)
2316 // (Only present if unbounded upper limit on loop)
2317 int32_t dataSize = fIntervalUpper < 0 ? 2 : 1;
2318 int32_t counterLoc = allocateStackData(dataSize);
2319
2320 int32_t op = buildOp(InitOp, counterLoc);
2321 fRXPat->fCompiledPat->setElementAt(op, topOfBlock);
2322
2323 // The second operand of CTR_INIT is the location following the end of the loop.
2324 // Must put in as a URX_RELOC_OPRND so that the value will be adjusted if the
2325 // compilation of something later on causes the code to grow and the target
2326 // position to move.
2327 int32_t loopEnd = fRXPat->fCompiledPat->size();
2328 op = buildOp(URX_RELOC_OPRND, loopEnd);
2329 fRXPat->fCompiledPat->setElementAt(op, topOfBlock+1);
2330
2331 // Followed by the min and max counts.
2332 fRXPat->fCompiledPat->setElementAt(fIntervalLow, topOfBlock+2);
2333 fRXPat->fCompiledPat->setElementAt(fIntervalUpper, topOfBlock+3);
2334
2335 // Apend the CTR_LOOP op. The operand is the location of the CTR_INIT op.
2336 // Goes at end of the block being looped over, so just append to the code so far.
2337 appendOp(LoopOp, topOfBlock);
2338
2339 if ((fIntervalLow & 0xff000000) != 0 ||
2340 (fIntervalUpper > 0 && (fIntervalUpper & 0xff000000) != 0)) {
2341 error(U_REGEX_NUMBER_TOO_BIG);
2342 }
2343
2344 if (fIntervalLow > fIntervalUpper && fIntervalUpper != -1) {
2345 error(U_REGEX_MAX_LT_MIN);
2346 }
2347}

2351UBool RegexCompile::compileInlineInterval() {
2352 if (fIntervalUpper > 10 || fIntervalUpper < fIntervalLow) {
2353 // Too big to inline. Fail, which will cause looping code to be generated.
2354 // (Upper < Lower picks up unbounded upper and errors, both.)
2355 return FALSE0;
2356 }
2357
2358 int32_t topOfBlock = blockTopLoc(FALSE0);
2359 if (fIntervalUpper == 0) {
2360 // Pathological case. Attempt no matches, as if the block doesn't exist.
2361 // Discard the generated code for the block.
2362 // If the block included parens, discard the info pertaining to them as well.
2363 fRXPat->fCompiledPat->setSize(topOfBlock);
2364 if (fMatchOpenParen >= topOfBlock) {
2365 fMatchOpenParen = -1;
2366 }
2367 if (fMatchCloseParen >= topOfBlock) {
2368 fMatchCloseParen = -1;
2369 }
2370 return TRUE1;
2371 }
2372
2373 if (topOfBlock != fRXPat->fCompiledPat->size()-1 && fIntervalUpper != 1) {
2374 // The thing being repeated is not a single op, but some
2375 // more complex block. Do it as a loop, not inlines.
2376 // Note that things "repeated" a max of once are handled as inline, because
2377 // the one copy of the code already generated is just fine.
2378 return FALSE0;
2379 }
2380
2381 // Pick up the opcode that is to be repeated
2382 //
2383 int32_t op = (int32_t)fRXPat->fCompiledPat->elementAti(topOfBlock);
2384
2385 // Compute the pattern location where the inline sequence
2386 // will end, and set up the state save op that will be needed.
2387 //
2388 int32_t endOfSequenceLoc = fRXPat->fCompiledPat->size()-1
2389 + fIntervalUpper + (fIntervalUpper-fIntervalLow);
2390 int32_t saveOp = buildOp(URX_STATE_SAVE, endOfSequenceLoc);
2391 if (fIntervalLow == 0) {
2392 insertOp(topOfBlock);
2393 fRXPat->fCompiledPat->setElementAt(saveOp, topOfBlock);
2394 }
2395
2396
2397
2398 // Loop, emitting the op for the thing being repeated each time.
2399 // Loop starts at 1 because one instance of the op already exists in the pattern,
2400 // it was put there when it was originally encountered.
2401 int32_t i;
2402 for (i=1; i<fIntervalUpper; i++ ) {
2403 if (i >= fIntervalLow) {
2404 appendOp(saveOp);
2405 }
2406 appendOp(op);
2407 }
2408 return TRUE1;
2409}

4403// Perform any pending (stacked) operations with precedence
4404// equal or greater to that of the next operator encountered
4405// in the expression.

4407void RegexCompile::setEval(int32_t nextOp) {
4408 UnicodeSet *rightOperand = NULL__null;
4409 UnicodeSet *leftOperand = NULL__null;
4410 for (;;) {
4411 U_ASSERT(fSetOpStack.empty()==FALSE);
4412 int32_t pendingSetOperation = fSetOpStack.peeki();
4413 if ((pendingSetOperation&0xffff0000) < (nextOp&0xffff0000)) {
4414 break;
4415 }
4416 fSetOpStack.popi();
4417 U_ASSERT(fSetStack.empty() == FALSE);
4418 rightOperand = (UnicodeSet *)fSetStack.peek();
4419 switch (pendingSetOperation) {
4420 case setNegation:
4421 rightOperand->complement();
4422 break;
4423 case setCaseClose:
4424 // TODO: need a simple close function. Ticket 6065
4425 rightOperand->closeOver(USET_CASE_INSENSITIVE);
4426 rightOperand->removeAllStrings();
4427 break;
4428 case setDifference1:
4429 case setDifference2:
4430 fSetStack.pop();
4431 leftOperand = (UnicodeSet *)fSetStack.peek();
4432 leftOperand->removeAll(*rightOperand);
4433 delete rightOperand;
4434 break;
4435 case setIntersection1:
4436 case setIntersection2:
4437 fSetStack.pop();
4438 leftOperand = (UnicodeSet *)fSetStack.peek();

4440 delete rightOperand;
4441 break;

4443 fSetStack.pop();
